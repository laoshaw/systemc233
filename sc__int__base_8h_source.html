20/*****************************************************************************
21
22 sc_int_base.h -- A signed integer whose length is less than 64 bit.
23
24 Unlike arbitrary precision, arithmetic and bitwise operations
25 are performed using the native types (hence capped at 64 bits).
26 The sc_int integer is useful when the user does not need
27 arbitrary precision and the performance is superior to
28 sc_bigint/sc_biguint.
29
30 Original Author: Amit Rao, Synopsys, Inc.

186 virtual bool concat_get_ctrl( sc_digit* dst_p, int low_i ) const
187 {
188 int bit_mask = 1 << (low_i % BITS_PER_DIGIT);
189 int word_i = low_i / BITS_PER_DIGIT;
190
191 dst_p[word_i] &= ~bit_mask;
192 return false;
193 }
194 virtual bool concat_get_data( sc_digit* dst_p, int low_i ) const
195 {
196 bool non_zero;
197 int bit_mask = 1 << (low_i % BITS_PER_DIGIT);
198 int word_i = low_i / BITS_PER_DIGIT;
199
200 if ( operator uint64() )
201 {
202 dst_p[word_i] |= bit_mask;
203 non_zero = true;
204 }
205 else
206 {
207 dst_p[word_i] &= ~bit_mask;
208 non_zero = false;
209 }
210 return non_zero;
211 }
212 virtual uint64 concat_get_uint64() const
213 { return operator uint64(); }

231 { return operator uint64(); }
232

237 { os << to_bool(); }

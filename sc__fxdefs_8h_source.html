250#define SC_ERROR_IF_IMPL_(cnd,id,msg) \
251 do { \
252 if( cnd ) { \
253 SC_REPORT_ERROR( id, msg ); \
254 sc_core::sc_abort(); /* can't recover from here */ \
255 } \
256 } while( false )
257
258#ifdef DEBUG_SYSTEMC
259# define SC_ASSERT_(cnd,msg) \

261#else
262# define SC_ASSERT_(cnd,msg) (void(0))
263#endif

56//------------------------------------------------------------------------------
57//"sc_process_b::add_static_event"
58//
59// This method adds an event to the list of static events, and sets the
60// event up to call back this process when it fires.
61//------------------------------------------------------------------------------
62void sc_process_b::add_static_event( const sc_event& e )
63{
64 sc_method_handle method_h; // This process as a method.
65 sc_thread_handle thread_h; // This process as a thread.
66
67
68 // CHECK TO SEE IF WE ARE ALREADY REGISTERED WITH THE EVENT:
69
70 for( int i = m_static_events.size() - 1; i >= 0; -- i ) {
71 if( &e == m_static_events[i] ) {
72 return;
73 }
74 }
75
76 // REMEMBER THE EVENT AND THEN REGISTER OUR OBJECT INSTANCE WITH IT:
77
78 m_static_events.push_back( &e );
79
80 switch ( m_process_kind )
81 {
82 case SC_THREAD_PROC_:
83 case SC_CTHREAD_PROC_:
84 thread_h = static_cast<sc_thread_handle>( this );
85 e.add_static( thread_h );
86 break;
87 case SC_METHOD_PROC_:
88 method_h = static_cast<sc_method_handle>( this );
89 e.add_static( method_h );
90 break;
91 default:
92 sc_assert( false );
93 break;
94 }
95}

97//------------------------------------------------------------------------------
98//"sc_process_b::disconnect_process"
99//
100// This method removes this object instance from use. It will be called by
101// the kill_process() methods of classes derived from it. This object instance
102// will be removed from any data structures it resides, other than existence.
103//------------------------------------------------------------------------------
104void sc_process_b::disconnect_process()
105{
106 int mon_n; // monitor queue size.
107 sc_thread_handle thread_h; // This process as a thread.
108
109 // IF THIS OBJECT IS PINING FOR THE FJORDS WE ARE DONE:
110
111 if ( m_state & ps_bit_zombie ) return;
112
113 // IF THIS IS A THREAD SIGNAL ANY MONITORS WAITING FOR IT TO EXIT:
114
115 switch ( m_process_kind )
116 {
117 case SC_THREAD_PROC_:
118 case SC_CTHREAD_PROC_:
119 thread_h = static_cast<sc_thread_handle>(this);
120 mon_n = thread_h->m_monitor_q.size();
121 if ( mon_n )
122 {
123 for ( int mon_i = 0; mon_i < mon_n; mon_i++ )
124 {
125 thread_h->m_monitor_q[mon_i]->signal( thread_h,
126 sc_process_monitor::spm_exit);
127 }
128 }
129 break;
130 default:
131 break;
132 }
133
134 // REMOVE EVENT WAITS, AND REMOVE THE PROCESS FROM ITS SC_RESET:
135
136 remove_dynamic_events();
137 remove_static_events();
138
139 for ( std::vector<sc_reset*>::size_type rst_i = 0; rst_i < m_resets.size(); rst_i++ )
140 {
141 m_resets[rst_i]->remove_process( this );
142 }
143 m_resets.resize(0);
144
145
146 // FIRE THE TERMINATION EVENT, MARK AS TERMINATED, AND DECREMENT THE COUNT:
147 //
148 // (1) We wait to set the process kind until after doing the removals
149 // above.
150 // (2) Decrementing the reference count will result in actual object
151 // deletion if we hit zero.
152
153 m_state = ps_bit_zombie;
154 if ( m_term_event_p ) m_term_event_p->notify();
155 reference_decrement();
156}

158//------------------------------------------------------------------------------
159//"sc_process_b::delete_process"
160//
161// This method deletes the current instance, if it is not the running
162// process. Otherwise, it is put in the simcontext's process deletion
163// queue.
164//
165// The reason for the two step deletion process is that the process from which
166// reference_decrement() is called may be the running process, so we may need
167// to wait until it goes idle.
168//------------------------------------------------------------------------------
169void sc_process_b::delete_process()
170{
171 sc_assert( m_references_n == 0 );
172
173 // Immediate deletion:
174
175 if ( NULL == sc_get_current_process_b() )
176 {
177 delete this;
178 }
179
180 // Deferred deletion: note we set the reference count to one for the call
181 // to reference_decrement that occurs in sc_simcontext::crunch().
182
183 else
184 {
185 m_references_n = 1;
186 detach();
187 simcontext()->mark_to_collect_process( this );
188 }
189}

202//------------------------------------------------------------------------------
203//"sc_process_b::dump_state"
204//
205// This method returns the process state as a string.
206//------------------------------------------------------------------------------
207std::string sc_process_b::dump_state() const
208{
209 std::string result;
210 result = "[";
211 if ( m_state == ps_normal )
212 {
213 result += " normal";
214 }
215 else
216 {
217 if ( m_state & ps_bit_disabled )
218 result += "disabled ";
219 if ( m_state & ps_bit_suspended )
220 result += "suspended ";
221 if ( m_state & ps_bit_ready_to_run )
222 result += "ready_to_run ";
223 if ( m_state & ps_bit_zombie )
224 result += "zombie ";
225 }
226 result += "]";
227 return result;
228}

243//------------------------------------------------------------------------------
244//"sc_process_b::remove_dynamic_events"
245//
246// This method removes this object instance from the events in its dynamic
247// event lists.
248//
249// Arguments:
250// skip_timeout = skip cleaning up the timeout event, it will be done
251// by sc_event_notify().
252//------------------------------------------------------------------------------
253void
254sc_process_b::remove_dynamic_events( bool skip_timeout )
255{
256 sc_method_handle method_h; // This process as a method.
257 sc_thread_handle thread_h; // This process as a thread.
258
259 m_trigger_type = STATIC;
260 switch ( m_process_kind )
261 {
262 case SC_THREAD_PROC_:
263 case SC_CTHREAD_PROC_:
264 thread_h = static_cast<sc_thread_handle>(this);
265 if ( thread_h->m_timeout_event_p && !skip_timeout ) {
266 thread_h->m_timeout_event_p->remove_dynamic(thread_h);
267 thread_h->m_timeout_event_p->cancel();
268 }
269 if ( m_event_p )
270 {
271 m_event_p->remove_dynamic( thread_h );
272 m_event_p = 0;
273 }
274 if ( m_event_list_p )
275 {
276 m_event_list_p->remove_dynamic( thread_h, 0 );
277 m_event_list_p->auto_delete();
278 m_event_list_p = 0;
279 }
280 break;
281 case SC_METHOD_PROC_:
282 method_h = static_cast<sc_method_handle>(this);
283 if ( method_h->m_timeout_event_p && !skip_timeout ) {
284 method_h->m_timeout_event_p->remove_dynamic(method_h);
285 method_h->m_timeout_event_p->cancel();
286 }
287 if ( m_event_p )
288 {
289 m_event_p->remove_dynamic( method_h );
290 m_event_p = 0;
291 }
292 if ( m_event_list_p )
293 {
294 m_event_list_p->remove_dynamic( method_h, 0 );
295 m_event_list_p->auto_delete();
296 m_event_list_p = 0;
297 }
298 break;
299 default: // Some other type, it needs to clean up itself.
300 // std::cout << "Check " << __FILE__ << ":" << __LINE__ << std::endl;
301 break;
302 }
303}

305//------------------------------------------------------------------------------
306//"sc_process_b::remove_static_events"
307//
308// This method removes this object instance from the events in its static
309// event list.
310//------------------------------------------------------------------------------
311void
312sc_process_b::remove_static_events()
313{
314 sc_method_handle method_h; // This process as a method.
315 sc_thread_handle thread_h; // This process as a thread.
316
317 switch ( m_process_kind )
318 {
319 case SC_THREAD_PROC_:
320 case SC_CTHREAD_PROC_:
321 thread_h = static_cast<sc_thread_handle>( this );
322 for( int i = m_static_events.size() - 1; i >= 0; -- i ) {
323 m_static_events[i]->remove_static( thread_h );
324 }
325 m_static_events.resize(0);
326 break;
327 case SC_METHOD_PROC_:
328 method_h = static_cast<sc_method_handle>( this );
329 for( int i = m_static_events.size() - 1; i >= 0; -- i ) {
330 m_static_events[i]->remove_static( method_h );
331 }
332 m_static_events.resize(0);
333 break;
334 default: // Some other type, it needs to clean up itself.
335 // std::cout << "Check " << __FILE__ << ":" << __LINE__ << std::endl;
336 break;
337 }
338}

375//------------------------------------------------------------------------------
376//"sc_process_b::reset_changed"
377//
378// This method is called when there is a change in the value of the
379// signal that was specified via reset_signal_is, or the value of the
380// m_sticky_reset field. We get called any time m_sticky_reset changes
381// or a signal value changes since, since we may need to throw an exception
382// or clear one. Note that this method may be called when there is no
383// active process, but rather the main simulator is executing so we must
384// check for that case.
385//
386// Arguments:
387// async = true if this is an asynchronous reset.
388// asserted = true if reset being asserted, false if being deasserted.
389//------------------------------------------------------------------------------
390void sc_process_b::reset_changed( bool async, bool asserted )
391{
392
393 // Error out on the corner case:
394
395 if ( !sc_allow_process_control_corners && !async &&
396 (m_state & ps_bit_suspended) )
397 {
398 report_error( SC_ID_PROCESS_CONTROL_CORNER_CASE_,
399 "synchronous reset changed on a suspended process" );
400 }
401
402 // IF THIS OBJECT IS PUSHING UP DAISIES WE ARE DONE:
403
404 if ( m_state & ps_bit_zombie ) return;
405
406 // Reset is being asserted:
407
408 if ( asserted )
409 {
410 // if ( m_reset_event_p ) m_reset_event_p->notify();
411 if ( async )
412 {
413 m_active_areset_n++;
414 if ( sc_is_running() ) throw_reset(true);
415 }
416 else
417 {
418 m_active_reset_n++;
419 if ( sc_is_running() ) throw_reset(false);
420 }
421 }
422
423 // Reset is being deasserted:
424
425 else
426 {
427 if ( async )
428 {
429 m_active_areset_n--;
430 }
431 else
432 {
433 m_active_reset_n--;
434 }
435 }
436
437 // Clear the throw type if there are no active resets.
438
439 if ( (m_throw_status == THROW_SYNC_RESET ||
440 m_throw_status == THROW_ASYNC_RESET) &&
441 m_active_areset_n == 0 && m_active_reset_n == 0 && !m_sticky_reset )
442 {
443 m_throw_status = THROW_NONE;
444 }
445}

462//------------------------------------------------------------------------------
463//"sc_process_b::reset_process"
464//
465// This inline method changes the reset state of this object instance and
466// conditionally its descendants.
467//
468// Notes:
469// (1) It is called for sync_reset_on() and sync_reset_off(). It is not used
470// for signal sensitive resets, though all reset flow ends up in
471// reset_changed().
472//
473// Arguments:
474// rt = source of the reset:
475// * reset_asynchronous - sc_process_handle::reset()
476// * reset_synchronous_off - sc_process_handle::sync_reset_off()
477// * reset_synchronous_on - sc_process_handle::sync_reset_on()
478// descendants = indication of how to process descendants.
479//------------------------------------------------------------------------------
480void sc_process_b::reset_process( reset_type rt,
481 sc_descendant_inclusion_info descendants )
482{
483
484 // PROCESS THIS OBJECT INSTANCE'S DESCENDANTS IF REQUESTED TO:
485
486 if ( descendants == SC_INCLUDE_DESCENDANTS )
487 {
488 const std::vector<sc_object*> children = get_child_objects();
489 int child_n = children.size();
490
491 for ( int child_i = 0; child_i < child_n; child_i++ )
492 {
493 sc_process_b* child_p = dynamic_cast<sc_process_b*>(children[child_i]);
494 if ( child_p ) child_p->reset_process(rt, descendants);
495 }
496 }
497
498 // PROCESS THIS OBJECT INSTANCE:
499
500 switch (rt)
501 {
502 // One-shot asynchronous reset: remove dynamic sensitivity and throw:
503 //
504 // If this is an sc_method only throw if it is active.
505
506 case reset_asynchronous:
507 if ( sc_get_status() != SC_RUNNING )
508 {
509 report_error(SC_ID_RESET_PROCESS_WHILE_NOT_RUNNING_);
510 }
511 else
512 {
513 remove_dynamic_events();
514 throw_reset(true);
515 }
516 break;
517
518 // Turn on sticky synchronous reset: use standard reset mechanism.
519
520 case reset_synchronous_on:
521 if ( m_sticky_reset == false )
522 {
523 m_sticky_reset = true;
524 reset_changed( false, true );
525 }
526 break;
527
528 // Turn off sticky synchronous reset: use standard reset mechanism.
529
530 default:
531 if ( m_sticky_reset == true )
532 {
533 m_sticky_reset = false;
534 reset_changed( false, false );
535 }
536 break;
537 }
538}

543// This is the object instance constructor for this class.

545sc_process_b::sc_process_b( const char* name_p, bool is_thread, bool free_host,
546 SC_ENTRY_FUNC method_p, sc_process_host* host_p,
547 const sc_spawn_options* /* opt_p */
548) :
549 sc_object( name_p ),
550 file(0),
551 lineno(0),
552 proc_id( simcontext()->next_proc_id()),
553 m_active_areset_n(0),
554 m_active_reset_n(0),
555 m_dont_init( false ),
556 m_dynamic_proc(),
557 m_event_p(0),
558 m_event_count(0),
559 m_event_list_p(0),
560 m_exist_p(0),
561 m_free_host( free_host ),
562 m_has_reset_signal( false ),
563 m_has_stack(false),
564 m_is_thread(is_thread),
565 m_last_report_p(0),
566 m_name_gen_p(0),
567 m_process_kind(SC_NO_PROC_),
568 m_references_n(1),
569 m_resets(),
570 m_reset_event_p(0),
571 m_resume_event_p(0),
572 m_runnable_p(0),
573 m_semantics_host_p( host_p ),
574 m_semantics_method_p ( method_p ),
575 m_state(ps_normal),
576 m_static_events(),
577 m_sticky_reset(false),
578 m_term_event_p(0),
579 m_throw_helper_p(0),
580 m_throw_status( THROW_NONE ),
581 m_timed_out(false),
582 m_timeout_event_p(0),
583 m_trigger_type(STATIC),
584 m_unwinding(false)
585{
586 // Check spawn phase: m_ready_to_simulate is set *after* elaboration_done()
587 unsigned spawned = SPAWN_ELAB;
588 spawned += simcontext()->elaboration_done(); // -> SPAWN_START, if true
589 spawned += simcontext()->m_ready_to_simulate; // -> SPAWN_SIM, if true
590 m_dynamic_proc = static_cast<spawn_t>(spawned);
591
592 // THIS OBJECT INSTANCE IS NOW THE LAST CREATED PROCESS:
593
594 m_last_created_process_p = this;
595 m_timeout_event_p = new sc_event( sc_event::kernel_event, "free_event" );
596}

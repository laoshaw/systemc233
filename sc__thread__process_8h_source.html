191//------------------------------------------------------------------------------
192//"sc_thread_process::suspend_me"
193//
194// This method suspends this object instance in favor of the next runnable
195// process. Upon awakening we check to see if an exception should be thrown.
196// There are two types of exceptions that can be thrown, synchronous reset
197// and asynchronous reset. At a future time there may be more asynchronous
198// exceptions. If an asynchronous reset is seen and there is not static reset
199// specified, or the static reset is not active then clear the throw
200// type for the next time this method is called.
201//
202// Notes:
203// (1) For an explanation of how the reset mechanism works see the top of
204// the file sc_reset.cpp.
205// (2) The m_sticky_reset field is used to handle synchronous resets that
206// are enabled via the sc_process_handle::sync_reset_on() method. These
207// resets are not generated by a signal, but rather are modal by
208// method call: sync_reset_on() - sync_reset_off().
209//------------------------------------------------------------------------------
210inline void sc_thread_process::suspend_me()
211{
212 // remember, if we're currently unwinding
213 bool unwinding_preempted = m_unwinding;
214
215 sc_simcontext* simc_p = simcontext();
216 sc_cor* cor_p = simc_p->next_cor();
217
218 // do not switch, if we're about to execute next (e.g. suicide)
219
220 if( m_cor_p != cor_p )
221 {
222 DEBUG_MSG( DEBUG_NAME , this, "suspending thread");
223 simc_p->cor_pkg()->yield( cor_p );
224 DEBUG_MSG( DEBUG_NAME , this, "resuming thread");
225 }
226
227 // IF THERE IS A THROW TO BE DONE FOR THIS PROCESS DO IT NOW:
228 //
229 // (1) Optimize THROW_NONE for speed as it is the normal case.
230 // (2) If this thread is already unwinding then suspend_me() was
231 // called from the catch clause to throw an exception on another
232 // process, so just go back to the catch clause.
233
234 if ( m_throw_status == THROW_NONE ) return;
235
236 if ( m_unwinding ) return;
237
238 switch( m_throw_status )
239 {
240 case THROW_ASYNC_RESET:
241 case THROW_SYNC_RESET:
242 DEBUG_MSG( DEBUG_NAME , this, "throwing reset for");
243 if ( m_reset_event_p ) m_reset_event_p->notify();
244 throw sc_unwind_exception( this, true );
245
246 case THROW_USER:
247 DEBUG_MSG( DEBUG_NAME, this, "invoking throw_it for");
248 m_throw_status = m_active_areset_n ? THROW_ASYNC_RESET :
249 (m_active_reset_n ? THROW_SYNC_RESET :
250 THROW_NONE);
251 m_throw_helper_p->throw_it();
252 break;
253
254 case THROW_KILL:
255 DEBUG_MSG( DEBUG_NAME, this, "throwing kill for");
256 throw sc_unwind_exception( this, false );
257
258 default: // THROWING_NOW
259 if( !unwinding_preempted )
260 SC_REPORT_FATAL( SC_ID_INTERNAL_ERROR_
261 , "unexpected unwinding/throw status" );
262 // may continue, if suppressed
263 m_throw_status = THROW_NONE;
264 DEBUG_MSG( DEBUG_NAME, this, "restarting thread");
265 break;
266 }
267}

373//------------------------------------------------------------------------------
374//"sc_thread_process::wait_cycles"
375//
376// This method suspends this object instance for the specified number of cycles.
377// A cycle is defined as the event the thread is set up to staticly wait on.
378// The field m_wait_cycle_n is set to one less than the number of cycles to
379// be waited for, since the value is tested before being decremented in
380// the simulation kernel.
381//------------------------------------------------------------------------------
382inline
383void
384sc_thread_process::wait_cycles( int n )
385{
386 if( m_unwinding )
387 SC_REPORT_ERROR( SC_ID_WAIT_DURING_UNWINDING_, name() );
388
389 if( n <= 0 )
390 SC_REPORT_ERROR( SC_ID_WAIT_NEGATIVE_CYCLES_, name() );
391
392 m_wait_cycle_n = n-1;
393 suspend_me();
394}

407inline
408void sc_thread_process::remove_monitor(sc_process_monitor* monitor_p)
409{
410 int mon_n = m_monitor_q.size();
411
412 for ( int mon_i = 0; mon_i < mon_n; mon_i++ )
413 {
414 if ( m_monitor_q[mon_i] == monitor_p )
415 {
416 m_monitor_q[mon_i] = m_monitor_q[mon_n-1];
417 m_monitor_q.resize(mon_n-1);
418 }
419 }

452//------------------------------------------------------------------------------
453//"sc_thread_process::trigger_static"
454//
455// This inline method adds the current thread to the queue of runnable
456// processes if required. This is the case if the following criteria
457// are met:
458// (1) The process is in a runnable state.
459// (2) The process is not already on the run queue.
460// (3) The process is expecting a static trigger,
461// dynamic event waits take priority.
462// (4) The process' static wait count is zero.
463//
464// If the triggering process is the same process, the trigger is
465// ignored as well, unless SC_ENABLE_IMMEDIATE_SELF_NOTIFICATIONS
466// is defined.
467//------------------------------------------------------------------------------
468inline
469void
470sc_thread_process::trigger_static()
471{
472 // No need to try queueing this thread if one of the following is true:
473 // (a) it is disabled
474 // (b) it is already queued for execution
475 // (c) it is waiting on a dynamic event
476 // (d) its wait count is not satisfied and it is not currently in reset
477
478 if ( (m_state & ps_bit_disabled) || is_runnable() ||
479 m_trigger_type != STATIC )
480 return;
481
482#if ! defined( SC_ENABLE_IMMEDIATE_SELF_NOTIFICATIONS )
483 if( SC_UNLIKELY_( sc_get_current_process_b() == this ) )
484 {
485 report_immediate_self_notification();
486 return;
487 }
488#endif // SC_ENABLE_IMMEDIATE_SELF_NOTIFICATIONS
489
490 if ( m_wait_cycle_n > 0 && THROW_NONE == m_throw_status )
491 {
492 --m_wait_cycle_n;
493 return;
494 }
495
496 // If we get here, then the thread has satisfied its wait criteria. If it is
497 // suspended, then mark its state as ready to run. If it is not suspended,
498 // then push it onto the runnable queue.
499
500 if ( m_state & ps_bit_suspended )
501 {
502 m_state = m_state | ps_bit_ready_to_run;
503 return;
504 }
505
506 simcontext()->push_runnable_thread(this);
507}

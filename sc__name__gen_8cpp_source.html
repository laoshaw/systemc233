46sc_name_gen::~sc_name_gen()
47{
48 sc_strhash<int*>::iterator it( m_unique_name_map );
49 for( ; ! it.empty(); it ++ ) {
50 delete it.contents();
51 }
52 m_unique_name_map.erase();
53}

56// to generate unique names for objects in an MT-Safe way
57
58const char*
59sc_name_gen::gen_unique_name( const char* basename_, bool preserve_first )
60{
61 if( basename_ == 0 || *basename_ == 0 ) {
62 SC_REPORT_ERROR( SC_ID_GEN_UNIQUE_NAME_, 0 );
63 basename_ = "unnamed"; // usually not reached
64 }
65 int* c = m_unique_name_map[basename_];
66 if( c == 0 ) {
67 c = new int( 0 );
68 m_unique_name_map.insert( const_cast<char*>( basename_ ), c );
69 if (preserve_first) {
70 m_unique_name = basename_;
71 } else {
72 std::stringstream sstr;
73 sstr << basename_ << "_" << *c;
74 sstr.str().swap( m_unique_name );
75 }
76 } else {
77 std::stringstream sstr;
78 sstr << basename_ << "_" << ++ (*c);
79 sstr.str().swap( m_unique_name );
80 }
81 return m_unique_name.c_str();
82}

45void
46halt( sc_simcontext* simc )
47{
48 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
49 switch( cpi->kind ) {
50 case SC_CTHREAD_PROC_: {
51 reinterpret_cast<sc_cthread_handle>( cpi->process_handle )->wait_halt();
52 break;
53 }
54 default:
55 SC_REPORT_ERROR( SC_ID_HALT_NOT_ALLOWED_, 0 );
56 break;
57 }
58}
59
60
61void
62wait( int n, sc_simcontext* simc )
63{
64 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
65 if( n <= 0 ) {
66 std::stringstream msg;
67 msg << "n = " << n;
68 SC_REPORT_ERROR( SC_ID_WAIT_N_INVALID_, msg.str().c_str() );
69 }
70 switch( cpi->kind ) {
71 case SC_THREAD_PROC_:
72 case SC_CTHREAD_PROC_:
73 reinterpret_cast<sc_thread_handle>( cpi->process_handle )->wait_cycles( n );
74 break;
75 default:
76 SC_REPORT_ERROR( SC_ID_WAIT_NOT_ALLOWED_, "\n "
77 "in SC_METHODs use next_trigger() instead" );
78 break;
79 }
80}

83void
84at_posedge( const sc_signal_in_if<bool>& s, sc_simcontext* simc )
85{
86 if( s.read() == true )
87 do { wait(simc); } while ( s.read() == true );
88 do { wait(simc); } while ( s.read() == false );
89}
90
91void
92at_posedge( const sc_signal_in_if<sc_dt::sc_logic>& s, sc_simcontext* simc )
93{
94 if( s.read() == '1' )
95 do { wait(simc); } while ( s.read() == '1' );
96 do { wait(simc); } while ( s.read() == '0' );
97}
98
99void
100at_negedge( const sc_signal_in_if<bool>& s, sc_simcontext* simc )
101{
102 if( s.read() == false )
103 do { wait(simc); } while ( s.read() == false );
104 do { wait(simc); } while ( s.read() == true );
105}
106
107void
108at_negedge( const sc_signal_in_if<sc_dt::sc_logic>& s, sc_simcontext* simc )
109{
110 if( s.read() == '0' )
111 do { wait(simc); } while ( s.read() == '0' );
112 do { wait(simc); } while ( s.read() == '1' );
113}

287template< typename ElementType
288 , typename AccessPolicy = sc_direct_access<ElementType> >
289class sc_vector_iter
290 : private AccessPolicy
291{
292 typedef ElementType element_type;
293 typedef typename AccessPolicy::policy access_policy;
294 typedef typename AccessPolicy::non_const_policy non_const_policy;
295 typedef typename AccessPolicy::const_policy const_policy;
296
297 typedef typename sc_meta::remove_const<ElementType>::type plain_type;
298 typedef const plain_type const_plain_type;
299 typedef typename sc_direct_access<plain_type>::const_policy
300 const_direct_policy;
301
302 friend class sc_vector< plain_type >;
303 template< typename, typename > friend class sc_vector_assembly;
304 template< typename, typename > friend class sc_vector_iter;
305
306 typedef sc_vector_iter this_type;
307 typedef sc_vector<plain_type> vector_type;
308 typedef sc_vector_base::storage_type storage_type;
309
310 // select correct base-type iterator
311 template< typename U > struct select_iter
312 { typedef typename storage_type::iterator type; };
313 template< typename U > struct select_iter< const U >
314 { typedef typename storage_type::const_iterator type; };
315
316 typedef typename select_iter<ElementType>::type raw_iterator;
317 typedef sc_vector_iter< const_plain_type, const_policy > const_iterator;
318 typedef sc_vector_iter< const_plain_type, const_direct_policy >
319 const_direct_iterator;
320
321 // underlying vector iterator
322
323 raw_iterator it_;
324
325 sc_vector_iter( raw_iterator it, access_policy acc = access_policy() )
326 : access_policy(acc), it_(it) {}
327
328 access_policy const & get_policy() const { return *this; }
329
330public:
331 // interface for Random Access Iterator category,
332 // see ISO/IEC 14882:2003(E), 24.1 [lib.iterator.requirements]
333
334 typedef std::random_access_iterator_tag iterator_category;
335 typedef std::ptrdiff_t difference_type;
336 typedef typename access_policy::type value_type;
337 typedef value_type & reference;
338 typedef value_type * pointer;
339
340 sc_vector_iter() : access_policy(), it_() {}
341
342 // iterator conversions to more const, and/or direct iterators
343 //
344 // Note: There is a minor risk to match unrelated classes (i.e. not sc_vector_iter<T,POL>),
345 // but MSVC 2005 does not correctly consider a restricted conversion constructor
346 // sc_vector_iter( const sc_vector_iter<OtherType,OtherPolicy>, SC_ENABLE_IF_ ...).
347 // To reduce this risk, the types used in the enable-if condition could be further
348 // tailored towards sc_vector(_iter), should the need arise.
349 //
350 // See also: sc_direct_access conversion constructor
351 template< typename It >
352 sc_vector_iter( const It& it
353 , SC_ENABLE_IF_((
354 sc_meta::is_more_const< element_type
355 , typename It::policy::element_type >
356 )) )
357 : access_policy( it.get_policy() ), it_( it.it_ )
358 {}
359
360 // step
361 this_type& operator++(){ ++it_; return *this; }
362 this_type& operator--(){ --it_; return *this; }
363 this_type operator++(int){ this_type old(*this); ++it_; return old; }
364 this_type operator--(int){ this_type old(*this); --it_; return old; }
365
366 // advance
367 this_type operator+( difference_type n ) const
368 { return this_type( it_ + n, get_policy()); }
369 this_type operator-( difference_type n ) const
370 { return this_type( it_ - n, get_policy()); }
371
372 this_type& operator+=( difference_type n ) { it_+=n; return *this; }
373 this_type& operator-=( difference_type n ) { it_-=n; return *this; }
374
375 // relations
376 bool operator==( const const_direct_iterator& that ) const
377 { return it_ == that.it_; }
378 bool operator!=( const const_direct_iterator& that ) const
379 { return it_ != that.it_; }
380 bool operator<=( const const_direct_iterator& that ) const
381 { return it_ <= that.it_; }
382 bool operator>=( const const_direct_iterator& that ) const
383 { return it_ >= that.it_; }
384 bool operator< ( const const_direct_iterator& that ) const
385 { return it_ < that.it_; }
386 bool operator> ( const const_direct_iterator& that ) const
387 { return it_ > that.it_; }
388
389 // dereference
390 reference operator*() const
391 { return *access_policy::get( static_cast<element_type*>((void*)*it_) ); }
392 pointer operator->() const
393 { return access_policy::get( static_cast<element_type*>((void*)*it_) ); }
394 reference operator[]( difference_type n ) const
395 { return *access_policy::get( static_cast<element_type*>((void*)it_[n]) ); }
396
397 // distance
398 difference_type operator-( const_direct_iterator const& that ) const
399 { return it_-that.it_; }
400
401}; // sc_vector_iter

669template< typename T >
670template< typename Creator >
671void
672sc_vector<T>::init( size_type n, Creator c )
673{
674 if ( base_type::check_init(n) )
675 {
676 // restore SystemC hierarchy context, if needed
677 sc_vector_base::context_scope scope( this );
678
679 base_type::reserve( n );
680 try
681 {
682 for ( size_type i = 0; i<n; ++i )
683 {
684 // this workaround is needed for SystemC 2.2/2.3 sc_bind
685 std::string name = make_name( basename(), i );
686 const char* cname = name.c_str();
687
688 element_type* p = c( cname, i ) ; // call Creator
689 base_type::push_back(p);
690 }
691 }
692 catch ( ... )
693 {
694 clear();
695 throw;
696 }
697 }
698}

707 delete &( (*this)[i] );
708 }
709 base_type::clear();

712template< typename Container, typename ArgumentIterator >
713typename Container::iterator
714sc_vector_do_bind( Container & cont
715 , ArgumentIterator first
716 , ArgumentIterator last
717 , typename Container::iterator from )
718{
719 typename Container::iterator end = cont.end();
720
721 if( !cont.size() || from == end || first == last )
722 cont.report_empty_bind( cont.kind(), from == end );
723
724 while( from!=end && first != last )
725 (*from++).bind( *first++ );
726 return from;
727}

729template< typename Container, typename ArgumentIterator >
730typename Container::iterator
731sc_vector_do_operator_paren( Container& cont
732 , ArgumentIterator first
733 , ArgumentIterator last
734 , typename Container::iterator from )
735{
736 typename Container::iterator end = cont.end();
737
738 if( !cont.size() || from == end || first == last )
739 cont.report_empty_bind( cont.kind(), from == end );
740
741 while( from!=end && first != last )
742 (*from++)( *first++ );
743 return from;
744}

752template< typename T, typename MT >
753std::vector< sc_object* > const &
754sc_vector_assembly<T,MT>::get_elements() const
755{
756 if( !child_vec_ )
757 child_vec_ = new std::vector< sc_object* >;
758
759 if( child_vec_->size() || !size() )
760 return *child_vec_;
761
762 child_vec_->reserve( size() );
763 for( const_iterator it=begin(); it != end(); ++it )
764 if( sc_object * obj = object_cast( const_cast<MT*>(&*it) ) )
765 child_vec_->push_back( obj );

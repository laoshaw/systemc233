64int
65sc_elab_and_sim( int argc, char* argv[] )
66{
67 int status = 1;
68 argc_orig = argc;
69 argv_orig = argv;
70
71 // Copy argv into a new structure to prevent sc_main from modifying the
72 // result returned from sc_argv.
73 std::vector<char*> argv_copy(argc + 1, static_cast<char*>(NULL));
74 for ( int i = 0; i < argc; ++i ) {
75 std::size_t size = std::strlen(argv[i]) + 1;
76 argv_copy[i] = new char[size];
77 std::copy(argv[i], argv[i] + size, argv_copy[i]);
78 }
79
80 try
81 {
82 pln();
83
84 // Perform initialization here
85 sc_in_action = true;
86
87 // copy array of pointers to keep allocated pointers for later release
88 std::vector<char*> argv_call = argv_copy;
89 status = sc_main( argc, &argv_call[0] );
90
91 // Perform cleanup here
92 sc_in_action = false;
93 }
94 catch( const sc_report& x )
95 {
96 sc_report_handler::get_handler()
97 ( x, sc_report_handler::get_catch_actions() );
98 }
99 catch( ... )
100 {
101 // translate other escaping exceptions
102 sc_report* err_p = sc_handle_exception();
103 if( err_p )
104 sc_report_handler::get_handler()
105 ( *err_p, sc_report_handler::get_catch_actions() );
106 delete err_p;
107 }
108
109 for ( int i = 0; i < argc; ++i ) {
110 delete[] argv_copy[i];
111 }
112
113 // IF DEPRECATION WARNINGS WERE ISSUED TELL THE USER HOW TO TURN THEM OFF
114
115 if ( sc_report_handler::get_count( SC_ID_IEEE_1666_DEPRECATION_ ) > 0 )
116 {
117 std::stringstream ss;
118
119 const char MSGNL[] = "\n ";
120 const char CODENL[] = "\n ";
121
122 ss << "You can turn off warnings about" << MSGNL
123 << "IEEE 1666 deprecated features by placing this method call" << MSGNL
124 << "as the first statement in your sc_main() function:\n" << CODENL
125 << "sc_core::sc_report_handler::set_actions( "
126 << "\"" << SC_ID_IEEE_1666_DEPRECATION_ << "\"," << CODENL
127 << " " /* indent param */
128 << "sc_core::SC_DO_NOTHING );"
129 << std::endl;
130
131 SC_REPORT_INFO( SC_ID_IEEE_1666_DEPRECATION_, ss.str().c_str() );
132 }
133
134 return status;
135}

64static inline
65unsigned
66next_pow2_index( std::size_t size )
67{
68 unsigned index = scfx_find_msb( size );
69 // If this was a power of 2 we are one bucket too low.
70 if( ~ (UINT64_ONE << index) & size ) index ++;
71 // If this is a 64-bit machine and we are using 32-bit words go down
72 // one slot size, as all the slots are 2x in size.
73 if ( index != 0 && ( sizeof(word_list) != sizeof(word) ) )
74 {
75 index -= 1;
76 }
77 return index;
78}
79
80static word_list* free_words[32] = { 0 };
81
82word*
83scfx_mant::alloc_word( std::size_t size )
84{
85 const int ALLOC_SIZE = 128;
86
87 unsigned slot_index = next_pow2_index( size );
88 unsigned alloc_size = ( UINT64_ONE << slot_index );
89
90 word_list*& slot = free_words[slot_index];
91
92 if( ! slot )
93 {
94 slot = new word_list[ALLOC_SIZE * alloc_size];
95 unsigned i;
96 for( i = 0; i < alloc_size*(ALLOC_SIZE-1) ; i+=alloc_size )
97 {
98 slot[i].m_next_p = &slot[i+alloc_size];
99 }
100 slot[i].m_next_p = 0;
101 }
102
103 word* result = (word*)slot;
104 free_words[slot_index] = slot[0].m_next_p;
105 return result;
106}
107
108void
109scfx_mant::free_word( word* array, std::size_t size )
110{
111 if( array && size )
112 {
113 int slot_index = next_pow2_index( size );
114 word_list* wl_p = (word_list*)array;
115
116 wl_p->m_next_p = free_words[slot_index];
117 free_words[slot_index] = wl_p;
118 }
119}

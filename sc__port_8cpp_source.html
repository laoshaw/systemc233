272// bind interface to this port
273
274void
275sc_port_base::bind( sc_interface& interface_ )
276{
277 if( m_bind_info == 0 ) {
278 // cannot bind an interface after elaboration
279 report_error( SC_ID_BIND_IF_TO_PORT_, "simulation running" );
280 return;
281 }
282
283 m_bind_info->vec.push_back( new sc_bind_elem( &interface_ ) );
284
285 if( ! m_bind_info->has_parent ) {
286 // add (cache) the interface
287 add_interface( &interface_ );
288 m_bind_info->last_add ++;
289 }
290}

293// bind parent port to this port
294
295void
296sc_port_base::bind( this_type& parent_ )
297{
298 if( m_bind_info == 0 ) {
299 // cannot bind a parent port after elaboration
300 report_error( SC_ID_BIND_PORT_TO_PORT_, "simulation running" );
301 return;
302 }
303
304 if( &parent_ == this ) {
305 report_error( SC_ID_BIND_PORT_TO_PORT_, "same port" );
306 return;
307 }
308
309 // check if parent port is already bound to this port
310#if 0
311 for( int i = m_bind_info->size() - 1; i >= 0; -- i ) {
312 if( &parent_ == m_bind_info->vec[i]->parent ) {
313 report_error( SC_ID_BIND_PORT_TO_PORT_, "already bound" );
314 return;
315 }
316 }
317#endif //
318
319 m_bind_info->vec.push_back( new sc_bind_elem( &parent_ ) );
320 m_bind_info->has_parent = true;
321 parent_.m_bind_info->is_leaf = false;
322}

404// support methods
405
406int
407sc_port_base::first_parent()
408{
409 for( int i = 0; i < m_bind_info->size(); ++ i ) {
410 if( m_bind_info->vec[i]->parent != 0 ) {
411 return i;
412 }
413 }
414 return -1;
415}
416
417void
418sc_port_base::insert_parent( int i )
419{
420 std::vector<sc_bind_elem*>& vec = m_bind_info->vec;
421
422 this_type* parent = vec[i]->parent;
423
424
425 // IF OUR PARENT HAS NO BINDING THEN IGNORE IT:
426 //
427 // Note that the zeroing of the parent pointer must occur before this
428 // test
429
430 vec[i]->parent = 0;
431 if ( parent->m_bind_info->vec.size() == 0 ) return;
432
433 vec[i]->iface = parent->m_bind_info->vec[0]->iface;
434 int n = parent->m_bind_info->size() - 1;
435 if( n > 0 ) {
436 // resize the bind vector (by adding new elements)
437 for( int k = 0; k < n; ++ k ) {
438 vec.push_back( new sc_bind_elem() );
439 }
440 // move elements in the bind vector
441 for( int k = m_bind_info->size() - n - 1; k > i; -- k ) {
442 vec[k + n]->iface = vec[k]->iface;
443 vec[k + n]->parent = vec[k]->parent;
444 }
445 // insert parent interfaces into the bind vector
446 for( int k = i + 1; k <= i + n; ++ k ) {
447 vec[k]->iface = parent->m_bind_info->vec[k - i]->iface;
448 vec[k]->parent = 0;
449 }
450 }
451}

454// called when elaboration is done
455
456void
457sc_port_base::complete_binding()
458{
459 // IF BINDING HAS ALREADY BEEN COMPLETED IGNORE THIS CALL:
460
461 sc_assert( m_bind_info != 0 );
462 if( m_bind_info->complete ) {
463 return;
464 }
465
466 // COMPLETE BINDING OF OUR PARENT PORTS SO THAT WE CAN USE THAT INFORMATION:
467
468 int i = first_parent();
469 while( i >= 0 ) {
470 m_bind_info->vec[i]->parent->complete_binding();
471 insert_parent( i );
472 i = first_parent();
473 }
474
475 // LOOP OVER BINDING INFORMATION TO COMPLETE THE BINDING PROCESS:
476
477 int size;
478 for( int j = 0; j < m_bind_info->size(); ++ j ) {
479 sc_interface* iface = m_bind_info->vec[j]->iface;
480
481 // if the interface is zero this was for an unbound port.
482 if ( iface == 0 ) continue;
483
484 // add (cache) the interface
485 if( j > m_bind_info->last_add ) {
486 add_interface( iface );
487 }
488
489 // only register "leaf" ports (ports without children)
490 if( m_bind_info->is_leaf ) {
491 iface->register_port( *this, if_typename() );
492 }
493
494 // complete static sensitivity for methods
495 size = m_bind_info->method_vec.size();
496 for( int k = 0; k < size; ++ k ) {
497 sc_bind_ef* p = m_bind_info->method_vec[k];
498 const sc_event& event = ( p->event_finder != 0 )
499 ? p->event_finder->find_event(iface)
500 : iface->default_event();
501 p->handle->add_static_event( event );
502 }
503
504 // complete static sensitivity for threads
505 size = m_bind_info->thread_vec.size();
506 for( int k = 0; k < size; ++ k ) {
507 sc_bind_ef* p = m_bind_info->thread_vec[k];
508 const sc_event& event = ( p->event_finder != 0 )
509 ? p->event_finder->find_event(iface)
510 : iface->default_event();
511 p->handle->add_static_event( event );
512 }
513
514 }
515
516 // MAKE SURE THE PROPER NUMBER OF BINDINGS OCCURRED:
517 //
518 // Make sure there are enough bindings, and not too many.
519
520 int actual_binds = interface_count();
521
522 if ( actual_binds > m_bind_info->max_size() )
523 {
524 std::stringstream msg;
525 msg << actual_binds << " binds exceeds maximum of "
526 << m_bind_info->max_size() << " allowed";
527 report_error( SC_ID_COMPLETE_BINDING_, msg.str().c_str() );
528 // may continue, if suppressed
529 }
530 switch ( m_bind_info->policy() )
531 {
532 case SC_ONE_OR_MORE_BOUND:
533 if ( actual_binds < 1 ) {
534 report_error( SC_ID_COMPLETE_BINDING_, "port not bound" );
535 // may continue, if suppressed
536 }
537 break;
538 case SC_ALL_BOUND:
539 if ( actual_binds < m_bind_info->max_size() || actual_binds < 1 ) {
540 std::stringstream msg;
541 msg << actual_binds << " actual binds is less than required "
542 << m_bind_info->max_size();
543 report_error( SC_ID_COMPLETE_BINDING_, msg.str().c_str() );
544 // may continue, if suppressed
545 }
546 break;
547 default: // SC_ZERO_OR_MORE_BOUND:
548 break;
549 }
550
551
552 // CLEAN UP: FREE BINDING STORAGE:
553
554 free_binding();
555
556 m_bind_info->complete = true;
557}

559void
560sc_port_base::free_binding()
561{
562 if( !m_bind_info )
563 return;
564
565 int size = m_bind_info->method_vec.size();
566 for( int k = 0; k < size; ++ k ) {
567 delete m_bind_info->method_vec[k];
568 }
569 // clear and release storage
570 sc_bind_info::ef_vector().swap(m_bind_info->method_vec);
571
572 size = m_bind_info->thread_vec.size();
573 for( int k = 0; k < size; ++ k ) {
574 delete m_bind_info->thread_vec[k];
575 }
576 // clear and release storage
577 sc_bind_info::ef_vector().swap(m_bind_info->thread_vec);
578}

619//
620// Registry for all ports.
621// FOR INTERNAL USE ONLY!
622// ----------------------------------------------------------------------------
623
624void
625sc_port_registry::insert( sc_port_base* port_ )
626{
627 if( sc_is_running() ) {
628 port_->report_error( SC_ID_INSERT_PORT_, "simulation running" );
629 return;
630 }
631
632 if( m_simc->elaboration_done() ) {
633 port_->report_error( SC_ID_INSERT_PORT_, "elaboration done" );
634 return;
635 }
636
637#if defined(DEBUG_SYSTEMC)
638 // check if port_ is already inserted
639 for( int i = size() - 1; i >= 0; -- i ) {
640 if( port_ == m_port_vec[i] ) {
641 port_->report_error( SC_ID_INSERT_PORT_, "port already inserted" );
642 return;
643 }
644 }
645#endif
646
647 // append the port to the current module's vector of ports
648 sc_module* curr_module = m_simc->hierarchy_curr();
649 if( curr_module == 0 ) {
650 port_->report_error( SC_ID_PORT_OUTSIDE_MODULE_ );
651 return;
652 }
653 curr_module->append_port( port_ );
654
655 // insert
656 m_port_vec.push_back( port_ );
657}
658
659void
660sc_port_registry::remove( sc_port_base* port_ )
661{
662 int i;
663 for( i = size() - 1; i >= 0; -- i ) {
664 if( port_ == m_port_vec[i] ) {
665 break;
666 }
667 }
668 if( i == -1 ) {
669 port_->report_error( SC_ID_REMOVE_PORT_, "port not registered" );
670 return;
671 }
672
673 // remove
674 m_port_vec[i] = m_port_vec.back();
675 m_port_vec.pop_back();
676}

695// called when construction is done
696
697bool
698sc_port_registry::construction_done()
699{
700 if( size() == m_construction_done )
701 // nothing has been updated
702 return true;
703
704 for( int i = size()-1; i >= m_construction_done; --i ) {
705 m_port_vec[i]->construction_done();
706 }
707
708 m_construction_done = size();
709 return false;
710}

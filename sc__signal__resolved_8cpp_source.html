41// Note that we assume that two drivers driving the resolved signal to a 1 or
42// 0 is O.K. This might not be true for all technologies, but is certainly
43// true for CMOS, the predominant technology in use today.
44
45SC_API const sc_dt::sc_logic_value_t
46sc_logic_resolution_tbl[4][4] =
47{ // 0 1 Z X
48 { sc_dt::Log_0, sc_dt::Log_X, sc_dt::Log_0, sc_dt::Log_X }, // 0
49 { sc_dt::Log_X, sc_dt::Log_1, sc_dt::Log_1, sc_dt::Log_X }, // 1
50 { sc_dt::Log_0, sc_dt::Log_1, sc_dt::Log_Z, sc_dt::Log_X }, // Z
51 { sc_dt::Log_X, sc_dt::Log_X, sc_dt::Log_X, sc_dt::Log_X } // X
52};

55// ----------------------------------------------------------------------------
56// FUNCTION : sc_logic_resolve
57//
58// Resolution function for sc_dt::sc_logic.
59// ----------------------------------------------------------------------------
60
61// resolves sc_dt::sc_logic values and returns the resolved value
62
63static void
64sc_logic_resolve( sc_dt::sc_logic& result_,
65 const std::vector<sc_dt::sc_logic>& values_ )
66{
67 int sz = values_.size();
68
69 sc_assert( sz != 0 );
70
71 if( sz == 1 ) {
72 result_ = values_[0];
73 return;
74 }
75
76 sc_dt::sc_logic_value_t res = values_[0].value();
77 for( int i = sz - 1; i > 0 && res != sc_dt::Log_X; -- i ) {
78 res = sc_logic_resolution_tbl[res][values_[i].value()];
79 }
80 result_ = res;
81}

90// write the new value
91
92void
93sc_signal_resolved::write( const value_type& value_ )
94{
95 sc_process_b* cur_proc = sc_get_current_process_b();
96
97 bool value_changed = false;
98 bool found = false;
99
100 for( int i = m_proc_vec.size() - 1; i >= 0; -- i ) {
101 if( cur_proc == m_proc_vec[i] ) {
102 if( value_ != m_val_vec[i] ) {
103 m_val_vec[i] = value_;
104 value_changed = true;
105 }
106 found = true;
107 break;
108 }
109 }
110
111 if( ! found ) {
112 m_proc_vec.push_back( cur_proc );
113 m_val_vec.push_back( value_ );
114 value_changed = true;
115 }
116
117 if( value_changed ) {
118 request_update();
119 }
120}

57// SIGNAL WRITING POLICIES
58//
59// Note: if you add a new policy to the enum below you will need to add
60// an additional overload of sc_reset::reset_signal_is() for the sc_signal<bool>
61// instance. That will require changes to sysc/kernel/sc_reset.cpp and
62// sysc/kernel/sc_reset.h
63
64enum sc_writer_policy
65{
66 SC_ONE_WRITER = 0,
67 SC_MANY_WRITERS = 1,
68 SC_UNCHECKED_WRITERS = 3
69};

71// signal forward declaration

78struct SC_API sc_writer_policy_nocheck_write
79{
80 bool check_write( sc_object* /* target */, bool /* value_changed */ )
81 { return true; }
82 bool needs_update() const { return false; }
83 void update() {}
84};
85
86struct SC_API sc_writer_policy_check_write
87{
88 bool check_write( sc_object* target, bool value_changed );
89 bool needs_update() const { return m_delta_only; }
90 void update();
91
92private:
93 static bool only_delta();
94
95protected:
96 sc_writer_policy_check_write( bool delta_only = only_delta() )
97 : m_delta_only( delta_only ), m_writer_p() {}
98
99 const bool m_delta_only;
100 sc_process_handle m_writer_p;
101};
102
103struct SC_API sc_writer_policy_check_delta
104 : sc_writer_policy_check_write
105{
106 // bool write_check(sc_object*, bool); /* inherited */
107
108 // always force update phase to reset process
109 bool needs_update() const { return true; }
110
111 // reset current writer during update phase
112 void update() { sc_process_handle().swap( m_writer_p ); }
113
114protected:
115 sc_writer_policy_check_delta()
116 : sc_writer_policy_check_write(true) {}
117};

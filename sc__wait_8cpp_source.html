43void warn_cthread_wait()
44{
45 static bool warn_wait = true;
46 if ( warn_wait )
47 {
48 warn_wait = false;
49 SC_REPORT_INFO(SC_ID_IEEE_1666_DEPRECATION_,
50 "all waits except wait() and wait(N)\n" \
51 " are deprecated for SC_CTHREAD, " \
52 "use an SC_THREAD instead");
53 }
54}

56SC_API void
57wait( sc_simcontext* simc )
58{
59 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
60 switch( cpi->kind ) {
61 case SC_THREAD_PROC_:
62 case SC_CTHREAD_PROC_: {
63 reinterpret_cast<sc_thread_handle>( cpi->process_handle )->wait_cycles();
64 break;
65 }
66 default:
67 SC_REPORT_ERROR( SC_ID_WAIT_NOT_ALLOWED_, "\n "
68 "in SC_METHODs use next_trigger() instead" );
69 break;
70 }
71}

75SC_API void
76wait( const sc_event& e, sc_simcontext* simc )
77{
78 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
79 switch( cpi->kind ) {
80 case SC_THREAD_PROC_: {
81 reinterpret_cast<sc_thread_handle>( cpi->process_handle )->wait( e );
82 break;
83 }
84 case SC_CTHREAD_PROC_: {
85 warn_cthread_wait();
86 sc_cthread_handle cthread_h =
87 reinterpret_cast<sc_cthread_handle>( cpi->process_handle );
88 cthread_h->wait( e );
89 cthread_h->wait_cycles();
90 break;
91 }
92 default:
93 SC_REPORT_ERROR( SC_ID_WAIT_NOT_ALLOWED_, "\n "
94 "in SC_METHODs use next_trigger() instead" );
95 break;
96 }
97}

99SC_API void
100wait( const sc_event_or_list& el, sc_simcontext* simc )
101{
102 if( el.empty() ) {
103 SC_REPORT_ERROR( SC_ID_EVENT_LIST_FAILED_,
104 "wait() on empty event list not allowed" );
105 // may continue, if suppressed
106 }
107
108 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
109 switch( cpi->kind ) {
110 case SC_THREAD_PROC_: {
111 reinterpret_cast<sc_thread_handle>( cpi->process_handle )->wait( el );
112 break;
113 }
114 case SC_CTHREAD_PROC_: {
115 warn_cthread_wait();
116 SC_REPORT_INFO(SC_ID_IEEE_1666_DEPRECATION_,
117 "wait(event_list) is deprecated for SC_CTHREAD, use SC_THREAD");
118 sc_cthread_handle cthread_h =
119 reinterpret_cast<sc_cthread_handle>( cpi->process_handle );
120 cthread_h->wait( el );
121 cthread_h->wait_cycles();
122 break;
123 }
124 default:
125 SC_REPORT_ERROR( SC_ID_WAIT_NOT_ALLOWED_, "\n "
126 "in SC_METHODs use next_trigger() instead" );
127 break;
128 }

272SC_API void
273next_trigger( sc_simcontext* simc )
274{
275 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
276 if( cpi->kind == SC_METHOD_PROC_ ) {
277 reinterpret_cast<sc_method_handle>( cpi->process_handle )->clear_trigger();
278 } else {
279 SC_REPORT_ERROR( SC_ID_NEXT_TRIGGER_NOT_ALLOWED_, "\n "
280 "in SC_THREADs and SC_CTHREADs use wait() instead" );
281 }
282}

398SC_API bool
399timed_out( sc_simcontext* simc )
400{
401 static bool warn_timed_out=true;
402 if ( warn_timed_out )
403 {
404 warn_timed_out = false;
405 SC_REPORT_INFO(SC_ID_IEEE_1666_DEPRECATION_,
406 "timed_out() function is deprecated" );
407 }
408
409 sc_curr_proc_handle cpi = simc->get_curr_proc_info();
410 return cpi->process_handle->timed_out();
411}

129void sc_unsigned_bitref::concat_set(const sc_signed& src, int low_i)
130{
131 if ( low_i < src.length() )

134 m_obj_p->set(m_index, src<0);

137void sc_unsigned_bitref::concat_set(const sc_unsigned& src, int low_i)
138{

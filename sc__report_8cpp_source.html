48static void sc_deprecated_report_ids(const char* method)
49{
50 static bool warn_report_ids_deprecated=true;
51 if ( warn_report_ids_deprecated )
52 {
53 std::string message;
54 message = "integer report ids are deprecated, use string values: ";
55 message += method;
56 warn_report_ids_deprecated=false;
57 SC_REPORT_INFO(SC_ID_IEEE_1666_DEPRECATION_, message.c_str());
58 }
59}

61static char empty_str[] = "";
62static inline char * empty_dup(const char * p)
63{
64 if ( p && *p )
65 {
66 char* result;
67 std::size_t size = strlen(p)+1;
68 result = new char[size];
69 std::copy(p, p + size, result);
70 return result;
71 }
72 else
73 {
74 return empty_str;
75 }
76}

177static bool warnings_are_errors = false;
178static const char unknown_id[] = "unknown id";
179
180void sc_report_handler::report(sc_severity severity_,
181 int id_,
182 const char* msg_,
183 const char* file_,
184 int line_ )
185{
186 sc_msg_def * md = sc_report_handler::mdlookup(id_);
187
188 if ( !md )
189 {
190 md = sc_report_handler::add_msg_type(unknown_id);
191 md->id = id_;
192 }
193
194 if ( severity_ == SC_WARNING && warnings_are_errors )
195 severity_ = SC_ERROR;
196
197 sc_actions actions = execute(md, severity_);
198 sc_report rep(severity_, md, msg_, file_, line_);
199
200 if ( actions & SC_CACHE_REPORT )
201 cache_report(rep);
202
203 if ( severity_ == SC_ERROR )
204 actions |= SC_THROW;
205 else if ( severity_ == SC_FATAL )
206 actions |= SC_ABORT;
207
208 handler(rep, actions);
209}

211void sc_report::register_id( int id, const char* msg )
212{
213 sc_deprecated_report_ids("sc_report::register_id()");
214 if( id < 0 ) {
215 SC_REPORT_ERROR( SC_ID_REGISTER_ID_FAILED_,
216 "invalid report id" );
217 return;
218 }
219 if( msg == 0 ) {
220 SC_REPORT_ERROR( SC_ID_REGISTER_ID_FAILED_,
221 "invalid report message" );
222 return;
223 }
224 sc_msg_def * md = sc_report_handler::mdlookup(id);
225
226 if ( !md )
227 md = sc_report_handler::add_msg_type(msg);
228
229 if ( !md ) {
230 SC_REPORT_ERROR( SC_ID_REGISTER_ID_FAILED_,
231 "report_map insertion error" );
232 return;
233 }
234
235 if( md->id != -1 ) {
236 if( strcmp( msg, md->msg_type ) != 0 ) {
237 SC_REPORT_ERROR( SC_ID_REGISTER_ID_FAILED_,
238 "report id already exists" );
239 }
240 return;
241 }
242 md->id = id;
243}

253bool sc_report::is_suppressed( int id )
254{
255 sc_deprecated_report_ids("sc_report::is_suppressed()");
256 sc_msg_def* md = sc_report_handler::mdlookup(id);
257
258 return md ? md->actions == SC_DO_NOTHING: false; // only do-nothing set
259}

261void sc_report::suppress_id(int id_, bool suppress)
262{

264 sc_msg_def* md = sc_report_handler::mdlookup(id_);

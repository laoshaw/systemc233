64sc_trace_file_base::sc_trace_file_base( const char* name, const char* extension )
65 : sc_trace_file()
66#if SC_TRACING_PHASE_CALLBACKS_
67 , sc_object( sc_gen_unique_name("$$$$kernel_tracefile$$$$") )
68#endif
69 , fp(0)
70 , trace_unit_fs()
71 , kernel_unit_fs()
72 , timescale_set_by_user(false)
73 , filename_()
74 , initialized_(false)
75 , trace_delta_cycles_(false)
76{
77 if( !name || !*name ) {
78 SC_REPORT_ERROR( SC_ID_TRACING_FOPEN_FAILED_, "no name given" );
79 return;
80 } else {
81 std::stringstream ss;
82 ss << name << "." << extension;
83 ss.str().swap( filename_ );
84 }
85
86#if SC_TRACING_PHASE_CALLBACKS_ == 1
87 // remove from hierarchy
88 sc_object::detach();
89 // register regular (non-delta) callbacks
90 sc_object::register_simulation_phase_callback( SC_BEFORE_TIMESTEP );
91#else // explicitly register with simcontext
92 sc_get_curr_simcontext()->add_trace_file( this );
93#endif
94}

96sc_trace_file_base::~sc_trace_file_base()
97{
98 if( !is_initialized() )
99 SC_REPORT_WARNING( SC_ID_TRACING_CLOSE_EMPTY_FILE_, filename() );
100
101 if( fp )
102 fclose(fp);
103
104#if SC_TRACING_PHASE_CALLBACKS_ == 0
105 // unregister from simcontext
106 sc_get_curr_simcontext()->remove_trace_file( this );
107#endif
108}

110/*****************************************************************************/
111// simulation phase callback based trigger
112//
113// The tracing updates are triggered
114// (- at the end of the initialization phase [disabled for now])
115// - before an update of the simulation time
116// - before returning to sc_start (via sc_pause() or sc_stop())
117// - after an update phase (if delta cycles need to be traced)
118//
119#if SC_TRACING_PHASE_CALLBACKS_
120void
121sc_trace_file_base::simulation_phase_callback()
122{
123 // delta cycle is traced at the end of an update phase
124 cycle( simcontext()->get_status() == SC_END_OF_UPDATE );
125}
126#endif // SC_TRACING_PHASE_CALLBACKS_

135bool
136sc_trace_file_base::initialize()
137{
138 if( initialized_ )
139 return false;
140
141 initialized_ = true;
142
143 if( !tracing_initialized_ ) {
144 tracing_initialized_ = true;
145 bool running_regression = ( getenv( "SYSTEMC_REGRESSION" ) != NULL );
146
147 // hide some messages during regression
148 if( running_regression ) {
149 sc_report_handler::set_actions( SC_ID_TRACING_TIMESCALE_DEFAULT_
150 , SC_INFO, SC_DO_NOTHING );
151 }
152 }
153
154 // open trace file
155 if(!fp) open_fp();
156
157 sc_time_tuple kernel_res_tuple = sc_time_tuple(sc_get_time_resolution());
158 kernel_unit_fs = kernel_res_tuple.value() * unit_to_fs(kernel_res_tuple.unit());
159
160 // setup timescale
161 if( !timescale_set_by_user )
162 {
163 trace_unit_fs = kernel_unit_fs;
164
165 std::stringstream ss;
166 ss << sc_get_time_resolution() << " (" << filename_ << ")";
167 SC_REPORT_INFO( SC_ID_TRACING_TIMESCALE_DEFAULT_
168 , ss.str().c_str() );
169 }
170
171 // initialize derived tracing implementation class (VCD/WIF)
172 do_initialize();
173
174 return initialized_;
175}

177void
178sc_trace_file_base::open_fp()
179{
180 sc_assert( !fp && filename() );
181 fp = fopen( filename(), "w" );
182 if( !fp ) {
183 SC_REPORT_ERROR( SC_ID_TRACING_FOPEN_FAILED_, filename() );
184 sc_abort(); // can't recover from here
185 }
186}

188void
189sc_trace_file_base::delta_cycles( bool flag )
190{
191 trace_delta_cycles_ = flag;
192#if SC_TRACING_PHASE_CALLBACKS_
193 if( trace_delta_cycles_ ) {
194 sc_object::register_simulation_phase_callback( SC_END_OF_UPDATE );
195 } else {
196 sc_object::unregister_simulation_phase_callback( SC_END_OF_UPDATE );
197 }
198#endif
199}

201void
202sc_trace_file_base::set_time_unit( double v, sc_time_unit tu )
203{
204 if( initialized_ )
205 {
206 std::stringstream ss;
207 ss << filename_ << "\n"
208 "\tTimescale unit cannot be changed once tracing has begun.\n"
209 "\tTo change the scale, create a new trace file.";
210 SC_REPORT_ERROR( SC_ID_TRACING_ALREADY_INITIALIZED_
211 , ss.str().c_str() );
212 return;
213 }
214
215 timescale_set_by_user = true;
216 trace_unit_fs = static_cast<unit_type>(v * unit_to_fs(tu));
217
218 // EMIT ADVISORY MESSAGE ABOUT CHANGE IN TIME SCALE:
219 {
220 std::stringstream ss;
221 ss << fs_unit_to_str(trace_unit_fs)
222 << " (" << filename_ << ")";
223 SC_REPORT_INFO( SC_ID_TRACING_TIMESCALE_UNIT_, ss.str().c_str() );
224 }
225}

251int
252sc_trace_file_base::low_units_len() const {
253 sc_assert(has_low_units());
254 double max_low_units = static_cast<double>(kernel_unit_fs / trace_unit_fs);
255 return static_cast<int>(log10(max_low_units));
256}

259void
260sc_trace_file_base::timestamp_in_trace_units(unit_type &high, unit_type &low) const
261{
262 unit_type time_now = sc_time_stamp().value();
263 unit_type delta_now = sc_delta_count_at_current_time();
264
265 if (has_low_units()) {
266 unit_type max_low_units = kernel_unit_fs / trace_unit_fs;
267 low = 0;
268 high = time_now;
269
270 if (delta_cycles()) {
271 low += delta_now % max_low_units;
272 high += delta_now / max_low_units;
273 }
274
275 } else {
276 unit_type unit_divisor = trace_unit_fs / kernel_unit_fs;
277 low = time_now % unit_divisor;
278 high = time_now / unit_divisor;
279
280 if (delta_cycles())
281 high += delta_now;
282 }
283}

286sc_time::value_type
287sc_trace_file_base::unit_to_fs(sc_time_unit tu)
288{
289 switch ( tu )
290 {
291 case SC_FS: return UINT64_C(1);
292 case SC_PS: return UINT64_C(1000);
293 case SC_NS: return UINT64_C(1000000);
294 case SC_US: return UINT64_C(1000000000);
295 case SC_MS: return UINT64_C(1000000000000);
296 case SC_SEC:return UINT64_C(1000000000000000);
297 default:
298 SC_REPORT_ERROR( SC_ID_TRACING_INVALID_TIMESCALE_UNIT_,
299 "unknown unit value" );
300 return 0;
301 }
302}

304std::string
305sc_trace_file_base::fs_unit_to_str(sc_trace_file_base::unit_type tu)
306{
307 switch (tu)
308 {
309 case UINT64_C(1): return "1 fs";
310 case UINT64_C(10): return "10 fs";
311 case UINT64_C(100): return "100 fs";
312 case UINT64_C(1000): return "1 ps";
313 case UINT64_C(10000): return "10 ps";
314 case UINT64_C(100000): return "100 ps";
315 case UINT64_C(1000000): return "1 ns";
316 case UINT64_C(10000000): return "10 ns";
317 case UINT64_C(100000000): return "100 ns";
318 case UINT64_C(1000000000): return "1 us";
319 case UINT64_C(10000000000): return "10 us";
320 case UINT64_C(100000000000): return "100 us";
321 case UINT64_C(1000000000000): return "1 ms";
322 case UINT64_C(10000000000000): return "10 ms";
323 case UINT64_C(100000000000000): return "100 ms";
324 case UINT64_C(1000000000000000): return "1 sec";
325 case UINT64_C(10000000000000000): return "10 sec";
326 case UINT64_C(100000000000000000): return "100 sec";
327 default: /* fail below */ (void)0;
328 }
329
330 std::stringstream ss;
331 ss << "not a power of ten: " << tu << " fs";
332 SC_REPORT_ERROR( SC_ID_TRACING_INVALID_TIMESCALE_UNIT_,
333 ss.str().c_str() );
334 return "";
335}

337// obtain formatted time string
338SC_API std::string
339localtime_string()
340{
341 char buf[200];
342 time_t long_time;
343 time(&long_time);
344 struct tm* p_tm = localtime(&long_time);
345 strftime(buf, 199, "%b %d, %Y %H:%M:%S", p_tm);
346 return buf;
347}

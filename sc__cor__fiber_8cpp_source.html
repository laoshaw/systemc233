67// destructor
68
69sc_cor_fiber::~sc_cor_fiber()
70{
71 if( m_fiber != 0 ) {
72 PVOID cur_fiber = GetCurrentFiber();
73 if( m_fiber != cur_fiber && this != m_pkg->get_main() )
74 DeleteFiber( m_fiber );
75 }
76}

79// ----------------------------------------------------------------------------
80// CLASS : sc_cor_pkg_fiber
81//
82// Coroutine package class implemented with QuickThreads.
83// ----------------------------------------------------------------------------
84
85int sc_cor_pkg_fiber::instance_count = 0;
86
87
88// constructor
89
90sc_cor_pkg_fiber::sc_cor_pkg_fiber( sc_simcontext* simc )
91: sc_cor_pkg( simc )
92{
93 if( ++ instance_count == 1 ) {
94 // initialize the main coroutine
95 sc_assert( main_cor.m_fiber == 0 );
96 main_cor.m_fiber = ConvertThreadToFiber( 0 );
97 main_cor.m_pkg = this;
98
99 if( !main_cor.m_fiber && GetLastError() == ERROR_ALREADY_FIBER ) {
100 // conversion of current thread to fiber has failed, because
101 // someone else already converted the main thread to a fiber
102 // -> store current fiber
103 main_cor.m_fiber = GetCurrentFiber();
104 }
105 sc_assert( main_cor.m_fiber != 0 );
106
107# if defined(__GNUC__) && __USING_SJLJ_EXCEPTIONS__
108 // initialize the current coroutine
109 sc_assert( curr_cor == 0 );
110 curr_cor = &main_cor;
111# endif
112 }
113}
114
115
116// destructor
117
118sc_cor_pkg_fiber::~sc_cor_pkg_fiber()
119{
120 if( -- instance_count == 0 ) {
121 // cleanup the main coroutine
122 main_cor.m_fiber = 0;
123 main_cor.m_pkg = 0;
124# if defined(__GNUC__) && __USING_SJLJ_EXCEPTIONS__
125 // cleanup the current coroutine
126 curr_cor = 0;
127# endif
128 }
129}

132// create a new coroutine
133
134sc_cor*
135sc_cor_pkg_fiber::create( std::size_t stack_size, sc_cor_fn* fn, void* arg )
136{
137 sc_cor_fiber* cor = new sc_cor_fiber;
138 cor->m_pkg = this;
139 cor->m_stack_size = stack_size;
140 cor->m_fiber = CreateFiberEx( cor->m_stack_size / 2, cor->m_stack_size, 0,
141 (LPFIBER_START_ROUTINE) fn, arg );
142 return cor;
143}

146// yield to the next coroutine
147
148void
149sc_cor_pkg_fiber::yield( sc_cor* next_cor )
150{
151 sc_cor_fiber* new_cor = static_cast<sc_cor_fiber*>( next_cor );
152# if defined(__GNUC__) && __USING_SJLJ_EXCEPTIONS__
153 // Switch SJLJ exception handling function contexts
154 _Unwind_SjLj_Register(&curr_cor->m_eh);
155 _Unwind_SjLj_Unregister(&new_cor->m_eh);
156 curr_cor = new_cor;
157# endif
158 SwitchToFiber( new_cor->m_fiber );
159}

168# if defined(__GNUC__) && __USING_SJLJ_EXCEPTIONS__
169 // Switch SJLJ exception handling function contexts
170 _Unwind_SjLj_Register(&curr_cor->m_eh);
171 _Unwind_SjLj_Unregister(&new_cor->m_eh);
172 curr_cor = new_cor;
173# endif
174 SwitchToFiber( new_cor->m_fiber );
175}

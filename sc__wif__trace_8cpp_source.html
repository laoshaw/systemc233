101class wif_trace
102{
103public:
104
105 wif_trace(const std::string& name_, const std::string& wif_name_);
106
107 // Needs to be pure virtual as has to be defined by the particular
108 // type being traced
109 virtual void write(FILE* f) = 0;
110
111 virtual void set_width();
112
113 // Comparison function needs to be pure virtual too
114 virtual bool changed() = 0;
115
116 // Got to declare this virtual as this will be overwritten
117 // by one base class
118 virtual void print_variable_declaration_line(FILE* f);
119
120 virtual ~wif_trace();
121
122 const std::string name; // Name of the variable
123 const std::string wif_name; // Name of the variable in WIF file
124 const char* wif_type; // WIF data type
125 int bit_width;
126};
127
128
129wif_trace::wif_trace(const std::string& name_,
130 const std::string& wif_name_)
131 : name(name_), wif_name(wif_name_), wif_type(0), bit_width(-1)
132{
133 /* Intentionally blank */
134}
135
136void
137wif_trace::print_variable_declaration_line( FILE* f )
138{
139 if( bit_width < 0 )
140 {
141 std::stringstream ss;
142 ss << "'" << name << "' has < 0 bits";
143 SC_REPORT_ERROR( SC_ID_TRACING_OBJECT_IGNORED_
144 , ss.str().c_str() );
145 return;
146 }
147
148 std::fprintf( f, "declare %s \"%s\" %s ",
149 wif_name.c_str(), name.c_str(), wif_type );
150
151 if( bit_width > 0 ) {
152 std::fprintf( f, "0 %d ", bit_width - 1 );
153 }
154 std::fprintf( f, "variable ;\n" );
155 std::fprintf( f, "start_trace %s ;\n", wif_name.c_str() );
156}

158void
159wif_trace::set_width()
160{
161 /* Intentionally Blank, should be defined for each type separately */
162}

164wif_trace::~wif_trace()
165{
166 /* Intentionally Blank */
167}

181

185 sc_dt::uint64 mask;
186};
187
188
189wif_uint64_trace::wif_uint64_trace(const sc_dt::uint64& object_,
190 const std::string& name_,
191 const std::string& wif_name_,
192 int width_)
193: wif_trace(name_, wif_name_), object(object_), old_value(object_),
194 mask(static_cast<sc_dt::uint64>(-1))
195{
196 bit_width = width_;
197 if (bit_width < static_cast<int>(sizeof(sc_dt::uint64)*BITS_PER_BYTE))
198 mask = ~(mask << bit_width);
199 wif_type = "BIT";
200}
201
202
203bool wif_uint64_trace::changed()
204{
205 return object != old_value;
206}
207
208
209void wif_uint64_trace::write(FILE* f)
210{
211 char buf[1000];
212 int bitindex;
213
214 // Check for overflow
215 if ((object & mask) != object)
216 {
217 for (bitindex = 0; bitindex < bit_width; bitindex++)
218 {
219 buf[bitindex]='0';
220 }
221 }
222 else
223 {
224 sc_dt::uint64 bit_mask = 1;
225 bit_mask = bit_mask << (bit_width-1);
226 for (bitindex = 0; bitindex < bit_width; bitindex++)
227 {
228 buf[bitindex] = (object & bit_mask)? '1' : '0';
229 bit_mask = bit_mask >> 1;
230 }
231 }
232 buf[bitindex] = '\0';
233 std::fprintf(f, "assign %s \"%s\" ;\n", wif_name.c_str(), buf);
234 old_value = object;
235}

459void wif_sc_unsigned_trace::write(FILE* f)
460{
461 static std::vector<char> buf(1024);
462 typedef std::vector<char>::size_type size_t;
463
464 if ( buf.size() < static_cast<size_t>(object.length()) ) {
465 size_t sz = ( static_cast<size_t>(object.length()) + 4096 ) & (~static_cast<size_t>(4096-1));
466 std::vector<char>( sz ).swap( buf ); // resize without copying values
467 }
468 char *buf_ptr = &buf[0];
469
470 for(int bitindex = object.length() - 1; bitindex >= 0; --bitindex) {
471 *buf_ptr++ = "01"[object[bitindex].to_bool()];
472 }
473 *buf_ptr = '\0';
474 std::fprintf(f, "assign %s \"%s\" ;\n", wif_name.c_str(), &buf[0]);
475 old_value = object;

1766void
1767wif_trace_file::cycle(bool this_is_a_delta_cycle)
1768{
1769 unit_type now_units_high, now_units_low;
1770
1771 // Trace delta cycles only when enabled
1772 if (!delta_cycles() && this_is_a_delta_cycle) return;
1773 // When delta cycles enabled, value changes happen only during delta cycles
1774 if (delta_cycles() && !this_is_a_delta_cycle) return;
1775
1776 // Check for initialization
1777 if( initialize() ) {
1778 return;
1779 };
1780
1781 timestamp_in_trace_units(now_units_high, now_units_low);
1782
1783 unit_type delta_units_high, delta_units_low;
1784 bool time_advanced = true;
1785 if (now_units_low < previous_units_low) {
1786 unit_type max_low_units = kernel_unit_fs / trace_unit_fs;
1787 delta_units_low = max_low_units - previous_units_low + now_units_low;
1788
1789 if (now_units_high <= previous_units_high)
1790 time_advanced = false;
1791
1792 delta_units_high = now_units_high - 1 - previous_units_high;
1793 } else {
1794 delta_units_low = now_units_low - previous_units_low;
1795
1796 if (now_units_high < previous_units_high || ((now_units_high == previous_units_high) && delta_units_low == 0) )
1797 time_advanced = false;
1798
1799 delta_units_high = now_units_high - previous_units_high;
1800 }
1801
1802 if (!time_advanced) {
1803 std::stringstream ss;
1804 ss <<"\n\tThis can occur when delta cycle tracing is activated."
1805 <<"\n\tSome delta cycles at " << sc_time_stamp() << " are not shown in trace file."
1806 <<"\n\tUse 'tracefile->set_time_unit(double, sc_time_unit);' to increase the time resolution.";
1807 SC_REPORT_WARNING( SC_ID_TRACING_REVERSED_TIME_, ss.str().c_str() );
1808 return;
1809 }
1810
1811 bool time_printed = false;
1812 wif_trace* const* const l_traces = &traces[0];
1813 for (int i = 0; i < (int)traces.size(); i++) {
1814 wif_trace* t = l_traces[i];
1815 if(t->changed()){
1816 if(time_printed == false) {
1817
1818 std::stringstream ss;
1819
1820 if(has_low_units())
1821 ss << "delta_time " << delta_units_high << std::setfill('0')
1822 << std::setw(low_units_len()) << delta_units_low << " ;\n";
1823 else
1824 ss << "delta_time " << delta_units_high <<" ;\n";
1825
1826 std::fputs(ss.str().c_str(), fp);
1827
1828 time_printed = true;
1829 }
1830
1831 // Write the variable
1832 t->write(fp);
1833 }
1834 }
1835
1836 if(time_printed) {
1837 std::fprintf(fp, "\n"); // Put another newline
1838 // We update previous_time_units only when we print time because
1839 // this field stores the previous time that was printed, not the
1840 // previous time this function was called
1841 previous_units_high = now_units_high;
1842 previous_units_low = now_units_low;
1843 }
1844}

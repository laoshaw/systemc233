79 { SC_DIGIT_ZERO, ~SC_DIGIT_ZERO, SC_DIGIT_ZERO, ~SC_DIGIT_ZERO };

82 { SC_DIGIT_ZERO, SC_DIGIT_ZERO, ~SC_DIGIT_ZERO, ~SC_DIGIT_ZERO };

85void
86sc_lv_base::init( int length_, const sc_logic& init_value )
87{
88 // check the length
89 if( length_ <= 0 ) {
90 SC_REPORT_ERROR( sc_core::SC_ID_ZERO_LENGTH_, 0 );
91 sc_core::sc_abort(); // can't recover from here
92 }
93 // allocate memory for the data and control words
94 m_len = length_;
95 m_size = (m_len - 1) / SC_DIGIT_SIZE + 1;
96 m_data = new sc_digit[m_size * 2];
97 m_ctrl = m_data + m_size;
98 // initialize the bits to 'init_value'
99 sc_digit dw = data_array[init_value.value()];
100 sc_digit cw = ctrl_array[init_value.value()];
101 int sz = m_size;
102 for( int i = 0; i < sz; ++ i ) {
103 m_data[i] = dw;
104 m_ctrl[i] = cw;
105 }
106 clean_tail();
107}

110void
111sc_lv_base::assign_from_string( const std::string& s )
112{
113 // s must have been converted to bin
114 int len = m_len;
115 int s_len = s.length() - 1;
116 int min_len = sc_min( len, s_len );
117 int i = 0;
118 for( ; i < min_len; ++ i ) {
119 char c = s[s_len - i - 1];
120 set_bit( i, sc_logic::char_to_logic[(int)c] );
121 }
122 // if formatted, fill the rest with sign(s), otherwise fill with zeros
123 sc_logic_value_t fill = (s[s_len] == 'F' ? sc_logic_value_t( s[0] - '0' )
124 : sc_logic_value_t( 0 ));
125 for( ; i < len; ++ i ) {
126 set_bit( i, fill );
127 }
128}

148sc_lv_base::sc_lv_base( const sc_lv_base& a )

151 m_size( a.m_size ),
152 m_data( new sc_digit[m_size * 2] ),
153 m_ctrl( m_data + m_size )

160 }

175
176bool
177sc_lv_base::is_01() const
178{
179 int sz = m_size;
180 for( int i = 0; i < sz; ++ i ) {
181 if( m_ctrl[i] != 0 ) {
182 return false;
183 }
184 }
185 return true;

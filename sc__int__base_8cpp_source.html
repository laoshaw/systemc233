158bool sc_int_subref_r::concat_get_ctrl( sc_digit* dst_p, int low_i ) const
159{
160 int dst_i; // Word in dst_p now processing.
161 int end_i; // Highest order word in dst_p to process.
162 int high_i; // Index of high order bit in dst_p to set.
163 uint_type mask; // Mask for bits to extract or keep.
164
165 dst_i = low_i / BITS_PER_DIGIT;
166 high_i = low_i + (m_left-m_right);
167 end_i = high_i / BITS_PER_DIGIT;
168 mask = ~mask_int[m_left][m_right];
169
170
171 // PROCESS THE FIRST WORD:
172
173 dst_p[dst_i] = (sc_digit)(dst_p[dst_i] & mask);
174 switch ( end_i - dst_i )
175 {
176 // BITS ARE ACROSS TWO WORDS:
177
178 case 1:
179 dst_i++;
180 dst_p[dst_i] = 0;
181 break;
182
183 // BITS ARE ACROSS THREE WORDS:
184
185 case 2:
186 dst_i++;
187 dst_p[dst_i++] = 0;
188 dst_p[dst_i] = 0;
189 break;
190
191 // BITS ARE ACROSS FOUR WORDS:
192
193 case 3:
194 dst_i++;
195 dst_p[dst_i++] = 0;
196 dst_p[dst_i++] = 0;
197 dst_p[dst_i] = 0;
198 break;
199 }
200 return false;
201}

204bool sc_int_subref_r::concat_get_data( sc_digit* dst_p, int low_i ) const
205{
206 int dst_i; // Word in dst_p now processing.
207 int end_i; // Highest order word in dst_p to process.
208 int high_i; // Index of high order bit in dst_p to set.
209 int left_shift; // Left shift for val.
210 uint_type mask; // Mask for bits to extract or keep.
211 bool non_zero; // True if value inserted is non-zero.
212 uint_type val; // Selection value extracted from m_obj_p.
213
214 dst_i = low_i / BITS_PER_DIGIT;
215 left_shift = low_i % BITS_PER_DIGIT;
216 high_i = low_i + (m_left-m_right);
217 end_i = high_i / BITS_PER_DIGIT;
218 mask = ~mask_int[m_left][m_right];
219 val = (m_obj_p->m_val & mask) >> m_right;
220 non_zero = val != 0;
221
222
223 // PROCESS THE FIRST WORD:
224
225 mask = ~(~UINT_ZERO << left_shift);
226 dst_p[dst_i] = (sc_digit)((dst_p[dst_i] & mask) |
227 ((val << left_shift) & DIGIT_MASK));
228
229 switch ( end_i - dst_i )
230 {
231 // BITS ARE ACROSS TWO WORDS:
232
233 case 1:
234 dst_i++;
235 val >>= (BITS_PER_DIGIT-left_shift);
236 dst_p[dst_i] = (sc_digit)(val & DIGIT_MASK);
237 break;
238
239 // BITS ARE ACROSS THREE WORDS:
240
241 case 2:
242 dst_i++;
243 val >>= (BITS_PER_DIGIT-left_shift);
244 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
245 val >>= BITS_PER_DIGIT;
246 dst_p[dst_i] = (sc_digit)val;
247 break;
248
249 // BITS ARE ACROSS FOUR WORDS:
250
251 case 3:
252 dst_i++;
253 val >>= (BITS_PER_DIGIT-left_shift);
254 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
255 val >>= BITS_PER_DIGIT;
256 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
257 val >>= BITS_PER_DIGIT;
258 dst_p[dst_i] = (sc_digit)val;
259 break;
260 }
261 return non_zero;
262}

276{
277 int_type val = m_obj_p->m_val;
278 uint_type mask = mask_int[m_left][m_right];
279 val &= mask;

282 m_obj_p->extend_sign();
283 return *this;

542sc_int_base::operator = ( const char* a )
543{
544 if( a == 0 ) {

597bool
598sc_int_base::xor_reduce() const
599{
600 uint_type mask = ~UINT_ZERO;
601 uint_type val = m_val & (mask >> m_ulen);
602 int n = SC_INTWIDTH;
603 do {
604 n >>= 1;
605 mask >>= n;
606 val = ((val & (mask << n)) >> n) ^ (val & mask);
607 } while( n != 1 );
608 return ( val != uint_type( 0 ) );
609}

612bool sc_int_base::concat_get_ctrl( sc_digit* dst_p, int low_i ) const
613{
614 int dst_i; // Word in dst_p now processing.
615 int end_i; // Highest order word in dst_p to process.
616 int left_shift; // Left shift for val.
617 uint_type mask; // Mask for bits to extract or keep.
618
619 dst_i = low_i / BITS_PER_DIGIT;
620 left_shift = low_i % BITS_PER_DIGIT;
621 end_i = (low_i + (m_len-1)) / BITS_PER_DIGIT;
622
623 mask = ~(~UINT_ZERO << left_shift);
624 dst_p[dst_i] = (sc_digit)(dst_p[dst_i] & mask);
625 dst_i++;
626 for ( ; dst_i <= end_i; dst_i++ ) dst_p[dst_i] = 0;
627 return false;
628}

630//------------------------------------------------------------------------------
631//"sc_int_base::concat_get_data"
632//
633// This method transfers the value of this object instance to the supplied
634// array of sc_unsigned digits starting with the bit specified by low_i within
635// the array of digits.
636//
637// Notes:
638// (1) we don't worry about masking the high order data we transfer since
639// concat_get_data() is called from low order bit to high order bit. So
640// the bits above where we place ours will be filled in by someone else.
641//
642// dst_p -> array of sc_unsigned digits to be filled in.
643// low_i = first bit within dst_p to be set.
644//------------------------------------------------------------------------------
645bool sc_int_base::concat_get_data( sc_digit* dst_p, int low_i ) const
646{
647 int dst_i; // Word in dst_p now processing.
648 int end_i; // Highest order word in dst_p to process.
649 int high_i; // Index of high order bit in dst_p to set.
650 int left_shift; // Left shift for val.
651 uint_type mask; // Mask for bits to extract or keep.
652 bool non_zero; // True if value inserted is non-zero.
653 uint_type val; // Value for this object.
654
655 dst_i = low_i / BITS_PER_DIGIT;
656 left_shift = low_i % BITS_PER_DIGIT;
657 high_i = low_i + (m_len-1);
658 end_i = high_i / BITS_PER_DIGIT;
659 val = m_val;
660 non_zero = val != 0;
661
662 // MASK OFF DATA TO BE TRANSFERRED BASED ON WIDTH:
663
664 if ( m_len < 64 )
665 {
666 mask = ~(~UINT_ZERO << m_len);
667 val &= mask;
668 }
669
670 // PROCESS THE FIRST WORD:
671
672 mask = (~UINT_ZERO << left_shift);
673 dst_p[dst_i] = (sc_digit)((dst_p[dst_i] & ~mask) |
674 ((val <<left_shift) & DIGIT_MASK));
675 switch ( end_i - dst_i )
676 {
677 // BITS ARE ACROSS TWO WORDS:
678
679 case 1:
680 dst_i++;
681 val >>= (BITS_PER_DIGIT-left_shift);
682 dst_p[dst_i] = (sc_digit)val;
683 break;
684
685 // BITS ARE ACROSS THREE WORDS:
686
687 case 2:
688 dst_i++;
689 val >>= (BITS_PER_DIGIT-left_shift);
690 dst_p[dst_i++] = ((sc_digit)val) & DIGIT_MASK;
691 val >>= BITS_PER_DIGIT;
692 dst_p[dst_i] = (sc_digit)val;
693 break;
694
695 // BITS ARE ACROSS FOUR WORDS:
696
697 case 3:
698 dst_i++;
699 val >>= (BITS_PER_DIGIT-left_shift);
700 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
701 val >>= BITS_PER_DIGIT;
702 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
703 val >>= BITS_PER_DIGIT;
704 dst_p[dst_i] = (sc_digit)val;
705 break;
706 }
707 return non_zero;
708}

69void SC_API tprintf(sc_trace_file* tf, const char* format, ...)
70{
71 static char buffer[4096];
72 va_list ap;
73 va_start(ap, format);
74 (void) vsprintf(buffer, format, ap);
75 va_end(ap);
76 if (tf) tf->write_comment(buffer);

247SC_API void
248sc_trace( sc_trace_file* tf,
249 const unsigned int& object,
250 const std::string& name,
251 const char** enum_literals )
252{
253 static bool warn_sc_trace_literals=true;
254 if ( warn_sc_trace_literals )
255 {
256 warn_sc_trace_literals=false;
257 SC_REPORT_INFO(SC_ID_IEEE_1666_DEPRECATION_,
258 "tracing of enumerated literals is deprecated" );
259 }
260
261 if( tf ) tf->trace( object, name, enum_literals );
262}

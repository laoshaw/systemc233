40class SC_API sc_plist_elem {
41 friend class sc_plist_base_iter;
42 friend class sc_plist_base;
43
44private:
45 sc_plist_elem() : data(0), prev(0), next(0)
46 {}
47 sc_plist_elem( void* d, sc_plist_elem* p, sc_plist_elem* n ) :
48 data(d), prev(p), next(n)
49 {}
50 ~sc_plist_elem()
51 {}
52
53 static void* operator new(std::size_t sz) { return sc_mempool::allocate(sz); }
54 static void operator delete(void* p, std::size_t sz) { sc_mempool::release(p, sz); }
55
56 void* data;
57 sc_plist_elem* prev;
58 sc_plist_elem* next;
59};

94sc_plist_base::handle_t
95sc_plist_base::push_back( void* d )
96{
97 handle_t q = new sc_plist_elem( d, tail, 0 );
98 if (tail) {
99 tail->next = q;
100 tail = q;
101 }
102 else {
103 head = tail = q;
104 }
105 return q;
106}

108sc_plist_base::handle_t
109sc_plist_base::push_front( void* d )
110{
111 handle_t q = new sc_plist_elem( d, (sc_plist_elem*) 0, head );
112 if (head) {
113 head->prev = q;
114 head = q;
115 }
116 else {
117 head = tail = q;
118 }
119 return q;
120}

122void*
123sc_plist_base::pop_back()
124{
125 handle_t q = tail;
126 void* d = q->data;
127 tail = tail->prev;
128 delete q;
129 if (tail != 0) {
130 tail->next = 0;
131 }
132 else {
133 head = 0;
134 }
135 return d;
136}

138void*
139sc_plist_base::pop_front()
140{
141 handle_t q = head;
142 void* d = q->data;
143 head = head->next;
144 delete q;
145 if (head != 0) {
146 head->prev = 0;
147 }
148 else {
149 tail = 0;
150 }
151 return d;
152}

154sc_plist_base::handle_t
155sc_plist_base::insert_before( handle_t h, void* d )
156{
157 if (h == 0) {
158 return push_back(d);
159 }
160 else {
161 handle_t q = new sc_plist_elem( d, h->prev, h );
162 h->prev->next = q;
163 h->prev = q;
164 return q;
165 }
166}
167
168sc_plist_base::handle_t
169sc_plist_base::insert_after( handle_t h, void* d )
170{
171 if (h == 0) {
172 return push_front(d);
173 }
174 else {
175 handle_t q = new sc_plist_elem( d, h, h->next );
176 h->next->prev = q;
177 h->next = q;
178 return q;
179 }
180}

182void*
183sc_plist_base::remove( handle_t h )
184{
185 if (h == head)
186 return pop_front();
187 else if (h == tail)
188 return pop_back();
189 else {
190 void* d = h->data;
191 h->prev->next = h->next;
192 h->next->prev = h->prev;
193 delete h;
194 return d;
195 }
196}

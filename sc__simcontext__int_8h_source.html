64inline
65const char*
66sc_get_current_process_name( const char * if_empty = NULL )
67{
68 sc_process_b* active_p; // active process to get name of.
69 const char* result; // name of active process.
70
71 active_p = sc_get_curr_simcontext()->get_curr_proc_info()->process_handle;
72 if ( active_p )
73 result = active_p->name();
74 else
75 result = if_empty;
76 return result;
77}

79// We use m_current_writer rather than m_curr_proc_info.process_handle to
80// return the active process for sc_signal<T>::check_write since that lets
81// us turn it off a library compile time, and only incur the overhead at
82// the time of process switches rather than having to interrogate an
83// additional switch every time a signal is written.

85inline
86void
87sc_simcontext::set_curr_proc( sc_process_b* process_h )
88{
89 m_curr_proc_info.process_handle = process_h;
90 m_curr_proc_info.kind = process_h->proc_kind();
91 m_current_writer =
92 (m_write_check != SC_SIGNAL_WRITE_CHECK_DISABLE_) ? process_h : 0;
93}

119// +----------------------------------------------------------------------------
120// |"sc_simcontext::preempt_with"
121// |
122// | This method executes the supplied thread immediately, suspending the
123// | caller. After executing the supplied thread the caller's execution will
124// | be restored. It is used to allow a thread to immediately throw an
125// | exception, e.g., when the thread's kill_process() method was called.
126// | There are three cases to consider:
127// | (1) The caller is a method, e.g., murder by method.
128// | (2) The caller is another thread instance, e.g., murder by thread.
129// | (3) The caller is this thread instance, e.g., suicide.
130// |
131// | Arguments:
132// | thread_h -> thread to be executed.
133// +----------------------------------------------------------------------------
134inline
135void
136sc_simcontext::preempt_with( sc_thread_handle thread_h )
137{
138 sc_thread_handle active_p; // active thread or null.
139 sc_curr_proc_info caller_info; // process info for caller.
140
141 // Determine the active process and take the thread to be run off the
142 // run queue, if its there, since we will be explicitly causing its
143 // execution.
144
145 active_p = dynamic_cast<sc_thread_handle>(sc_get_current_process_b());
146 if ( thread_h->next_runnable() != NULL )
147 remove_runnable_thread( thread_h );
148
149 // THE CALLER IS A METHOD:
150 //
151 // (a) Set the current process information to our thread.
152 // (b) If the method was called by an invoker thread push that thread
153 // onto the front of the run queue, this will cause the method
154 // to be resumed after this thread waits.
155 // (c) Invoke our thread directly by-passing the run queue.
156 // (d) Restore the process info to the caller.
157 // (e) Check to see if the calling method should throw an exception
158 // because of activity that occurred during the preemption.
159
160 if ( active_p == NULL )
161 {
162 std::vector<sc_thread_handle>* invokers_p; // active invokers stack.
163 sc_thread_handle invoke_thread_p; // latest invocation thread.
164 sc_method_handle method_p; // active method.
165
166 method_p = dynamic_cast<sc_method_handle>(sc_get_current_process_b());
167 invokers_p = &get_active_invokers();
168 caller_info = m_curr_proc_info;
169 if ( invokers_p->size() != 0 )
170 {
171 invoke_thread_p = invokers_p->back();
172 DEBUG_MSG( DEBUG_NAME, invoke_thread_p,
173 "queueing invocation thread to execute next" );
174 execute_thread_next(invoke_thread_p);
175 }
176 DEBUG_MSG( DEBUG_NAME, thread_h, "preempting method with thread" );
177 set_curr_proc( (sc_process_b*)thread_h );
178 m_cor_pkg->yield( thread_h->m_cor_p );
179 m_curr_proc_info = caller_info;
180 DEBUG_MSG(DEBUG_NAME, thread_h, "back from preempting method w/thread");
181 method_p->check_for_throws();
182 }
183
184 // CALLER IS A THREAD, BUT NOT THE THREAD TO BE RUN:
185 //
186 // (a) Push the calling thread onto the front of the runnable queue
187 // so it be the first thread to be run after this thread.
188 // (b) Push the thread to be run onto the front of the runnable queue so
189 // it will execute when we suspend the calling thread.
190 // (c) Suspend the active thread.
191
192 else if ( active_p != thread_h )
193 {
194 DEBUG_MSG( DEBUG_NAME, thread_h,
195 "preempting active thread with thread" );
196 execute_thread_next( active_p );
197 execute_thread_next( thread_h );
198 active_p->suspend_me();
199 }
200
201 // CALLER IS THE THREAD TO BE RUN:
202 //
203 // (a) Push the thread to be run onto the front of the runnable queue so
204 // it will execute when we suspend the calling thread.
205 // (b) Suspend the active thread.
206
207 else
208 {
209 DEBUG_MSG(DEBUG_NAME,thread_h,"self preemption of active thread");
210 execute_thread_next( thread_h );
211 active_p->suspend_me();
212 }
213}

245inline
246sc_method_handle
247sc_simcontext::pop_runnable_method()
248{
249 sc_method_handle method_h = m_runnable->pop_method();
250 if( method_h == 0 ) {
251 reset_curr_proc();
252 return 0;
253 }
254 set_curr_proc( (sc_process_b*)method_h );
255 return method_h;
256}
257
258inline
259sc_thread_handle
260sc_simcontext::pop_runnable_thread()
261{
262 sc_thread_handle thread_h = m_runnable->pop_thread();
263 if( thread_h == 0 ) {
264 reset_curr_proc();
265 return 0;
266 }
267 set_curr_proc( (sc_process_b*)thread_h );
268 return thread_h;
269}

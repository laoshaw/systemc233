153inline bool operator == (
154 const sc_process_handle& left, const sc_process_handle& right )
155{
156 return (left.m_target_p != 0) && (right.m_target_p != 0) &&
157 (left.m_target_p == right.m_target_p);
158}
159
160inline bool operator != (
161 const sc_process_handle& left, const sc_process_handle& right )
162{
163 return (left.m_target_p == 0) || (right.m_target_p == 0) ||
164 (left.m_target_p != right.m_target_p);
165}

167inline bool operator < (
168 const sc_process_handle& left, const sc_process_handle& right )
169{
170 return left.m_target_p < right.m_target_p;

226//------------------------------------------------------------------------------
227//"sc_process_handle::operator ="
228//
229// This assignment operator signature is call by value rather than reference.
230// This means that an sc_process_handle instance will be created and the
231// target for that instance will be incremented before the assignment is done.
232// The assignment is done using the swap() method, which simply swaps the
233// targets of 'orig' and this object instance. We don't need to increment
234// the reference count for our new target since that was done when 'orig'
235// was created. Our old target's reference count will be decremented when
236// 'orig' is deleted.
237// orig = sc_process_handle object instance to be copied from.
238// Result is a reference for this object instance.
239//------------------------------------------------------------------------------
240inline sc_process_handle&
241sc_process_handle::operator = ( sc_process_handle orig )
242{
243 swap( orig );
244 return *this;
245}

252// the reference count for its target.

490//------------------------------------------------------------------------------
491//"sc_process_handle::sc_throw_it"
492//
493// This method throws the supplied exception to the process whose handle this
494// object instance is, and optionally to the process' descendants. Once the
495// exception is thrown the currently executed process will suspend to allow
496// the exception to be propagated. Once the propagation has occurred the
497// current process will be resumed.
498//
499// Notes:
500// (1) We allocate the helper function on the stack, see the description of
501// sc_throw_it<EXCEPT>, in sc_process.h, for why.
502//
503// Arguments:
504// exception = exception to be thrown
505// descendants = indication of whether descendant processes should also
506// receive the throw.
507//------------------------------------------------------------------------------
508template<typename EXCEPT>
509inline void sc_process_handle::throw_it( const EXCEPT& exception,
510 sc_descendant_inclusion_info descendants)
511{
512 sc_throw_it<EXCEPT> helper(exception); // helper to throw the exception.
513
514 if ( !m_target_p )
515 {
516 SC_REPORT_WARNING( SC_ID_EMPTY_PROCESS_HANDLE_, "throw_it()");
517 return;
518 }
519 m_target_p->throw_user(helper, descendants);
520}

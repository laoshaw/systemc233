38// simulation status codes
39
40const int SC_SIM_OK = 0;
41const int SC_SIM_ERROR = 1;
42const int SC_SIM_USER_STOP = 2;
43
44enum sc_status
45{ // sc_get_status values:
46 SC_UNITIALIZED=0x00, // initialize() not called yet
47
48 SC_ELABORATION = 0x01, // during module hierarchy construction
49 SC_BEFORE_END_OF_ELABORATION = 0x02, // during before_end_of_elaboration()
50 SC_END_OF_ELABORATION = 0x04, // during end_of_elaboration()
51 SC_START_OF_SIMULATION = 0x08, // during start_of_simulation()
52
53 SC_RUNNING = 0x10, // initialization, evaluation or update
54 SC_PAUSED = 0x20, // when scheduler stopped by sc_pause()
55 SC_STOPPED = 0x40, // when scheduler stopped by sc_stop()
56 SC_END_OF_SIMULATION = 0x80, // during end_of_simulation()
57
58 // detailed simulation phases (for dynamic callbacks)
59 SC_END_OF_INITIALIZATION = 0x100, // after initialization
60// SC_END_OF_EVALUATION = 0x200, // between eval and update
61 SC_END_OF_UPDATE = 0x400, // after update/notify phase
62 SC_BEFORE_TIMESTEP = 0x800, // before next time step
63
64 SC_STATUS_LAST = SC_BEFORE_TIMESTEP,
65 SC_STATUS_ANY = 0xdff
66};
67
68// pretty-printing of sc_status values
69SC_API std::ostream& operator << ( std::ostream&, sc_status );
70
71} // namespace sc_core

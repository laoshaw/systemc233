157bool sc_uint_subref_r::concat_get_ctrl( sc_digit* dst_p, int low_i ) const
158{
159 int dst_i; // Word in dst_p now processing.
160 int end_i; // Highest order word in dst_p to process.
161 int left_shift; // Left shift for val.
162 uint_type mask; // Mask for bits to extract or keep.
163
164 dst_i = low_i / BITS_PER_DIGIT;
165 left_shift = low_i % BITS_PER_DIGIT;
166 end_i = (low_i + (m_left-m_right)) / BITS_PER_DIGIT;
167
168 mask = ~(~UINT_ZERO << left_shift);
169 dst_p[dst_i] = (sc_digit)((dst_p[dst_i] & mask));
170
171 dst_i++;
172 for ( ; dst_i <= end_i; dst_i++ ) dst_p[dst_i] = 0;
173
174 return false;
175}
176
177bool sc_uint_subref_r::concat_get_data( sc_digit* dst_p, int low_i ) const
178{
179 int dst_i; // Word in dst_p now processing.
180 int end_i; // Highest order word in dst_p to process.
181 int high_i; // Index of high order bit in dst_p to set.
182 int left_shift; // Left shift for val.
183 uint_type mask; // Mask for bits to extract or keep.
184 bool result; // True if inserting non-zero value.
185 uint_type val; // Selection value extracted from m_obj_p.
186
187 dst_i = low_i / BITS_PER_DIGIT;
188 left_shift = low_i % BITS_PER_DIGIT;
189 high_i = low_i + (m_left-m_right);
190 end_i = high_i / BITS_PER_DIGIT;
191 mask = ~mask_int[m_left][m_right];
192 val = (m_obj_p->m_val & mask) >> m_right;
193 result = val != 0;
194
195
196 // PROCESS THE FIRST WORD:
197
198 mask = ~(~UINT_ZERO << left_shift);
199 dst_p[dst_i] = (sc_digit)(((dst_p[dst_i] & mask)) |
200 ((val << left_shift) & DIGIT_MASK));
201
202 switch ( end_i - dst_i )
203 {
204 // BITS ARE ACROSS TWO WORDS:
205
206 case 1:
207 dst_i++;
208 val >>= (BITS_PER_DIGIT-left_shift);
209 dst_p[dst_i] = (sc_digit)val;
210 break;
211
212 // BITS ARE ACROSS THREE WORDS:
213
214 case 2:
215 dst_i++;
216 val >>= (BITS_PER_DIGIT-left_shift);
217 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
218 val >>= BITS_PER_DIGIT;
219 dst_p[dst_i] = (sc_digit)val;
220 break;
221
222 // BITS ARE ACROSS THREE WORDS:
223
224 case 3:
225 dst_i++;
226 val >>= (BITS_PER_DIGIT-left_shift);
227 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
228 val >>= BITS_PER_DIGIT;
229 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
230 val >>= BITS_PER_DIGIT;
231 dst_p[dst_i] = (sc_digit)val;
232 break;
233 }
234 return result;
235}

247sc_uint_subref&
248sc_uint_subref::operator = ( uint_type v )
249{
250 uint_type val = m_obj_p->m_val;
251 uint_type mask = mask_int[m_left][m_right];
252 val &= mask;
253 val |= (v << m_right) & ~mask;
254 m_obj_p->m_val = val;
255 m_obj_p->extend_sign();
256 return *this;
257}

448{
449 int minlen = sc_min( m_len, a.length() );
450 int i = 0;
451 for( ; i < minlen; ++ i ) {
452 set( i, a.test( i ) );

454 bool sgn = a.sign();

456 // sign extension

512sc_uint_base&
513sc_uint_base::operator = ( const char* a )
514{
515 if( a == 0 ) {
516 SC_REPORT_ERROR( sc_core::SC_ID_CONVERSION_FAILED_,
517 "character string is zero" );
518 }
519 else if( *a == 0 ) {
520 SC_REPORT_ERROR( sc_core::SC_ID_CONVERSION_FAILED_,
521 "character string is empty" );
522 }
523 else try {
524 int len = m_len;
525 sc_ufix aa( a, len, len, SC_TRN, SC_WRAP, 0, SC_ON );
526 return this->operator = ( aa );
527 } catch( const sc_core::sc_report & ) {
528 std::stringstream msg;
529 msg << "character string '" << a << "' is not valid";
530 SC_REPORT_ERROR( sc_core::SC_ID_CONVERSION_FAILED_, msg.str().c_str() );
531 }
532 return *this;
533}

569bool
570sc_uint_base::xor_reduce() const
571{
572 uint_type mask = ~UINT_ZERO;
573 uint_type val = m_val;
574 int n = SC_INTWIDTH;
575 do {
576 n >>= 1;
577 mask >>= n;
578 val = ((val & (mask << n)) >> n) ^ (val & mask);
579 } while( n != 1 );
580 return ( val != uint_type( 0 ) );
581}

605//------------------------------------------------------------------------------
606//"sc_uint_base::concat_get_data"
607//
608// This method transfers the value of this object instance to the supplied
609// array of sc_unsigned digits starting with the bit specified by low_i within
610// the array of digits.
611//
612// Notes:
613// (1) we don't worry about masking the high order data we transfer since
614// concat_get_data() is called from low order bit to high order bit. So
615// the bits above where we place ours will be filled in by someone else.
616//
617// dst_p -> array of sc_unsigned digits to be filled in.
618// low_i = first bit within dst_p to be set.
619//------------------------------------------------------------------------------
620bool sc_uint_base::concat_get_data( sc_digit* dst_p, int low_i ) const
621{
622 int dst_i; // Word in dst_p now processing.
623 int end_i; // Highest order word in dst_p to process.
624 int high_i; // Index of high order bit in dst_p to set.
625 int left_shift; // Left shift for val.
626 uint_type mask; // Mask for bits to extract or keep.
627 bool result; // True if inserting non-zero value.
628 uint_type val; // Value for this object.
629
630 dst_i = low_i / BITS_PER_DIGIT;
631 left_shift = low_i % BITS_PER_DIGIT;
632 high_i = low_i + (m_len-1);
633 end_i = high_i / BITS_PER_DIGIT;
634 val = m_val;
635 result = val != 0;
636
637 // MASK OFF DATA TO BE TRANSFERRED BASE ON WIDTH:
638
639 if ( m_len < 64 )
640 {
641 mask = ~(~UINT_ZERO << m_len);
642 val &= mask;
643 }
644
645 // PROCESS THE FIRST WORD:
646
647 mask = ~(~UINT_ZERO << left_shift);
648 dst_p[dst_i] = (sc_digit)(((dst_p[dst_i] & mask)) |
649 ((val << left_shift) & DIGIT_MASK));
650
651 switch ( end_i - dst_i )
652 {
653 // BITS ARE ACROSS TWO WORDS:
654
655 case 1:
656 dst_i++;
657 val >>= (BITS_PER_DIGIT-left_shift);
658 dst_p[dst_i] = (sc_digit)val;
659 break;
660
661 // BITS ARE ACROSS THREE WORDS:
662
663 case 2:
664 dst_i++;
665 val >>= (BITS_PER_DIGIT-left_shift);
666 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
667 val >>= BITS_PER_DIGIT;
668 dst_p[dst_i] = (sc_digit)val;
669 break;
670
671 // BITS ARE ACROSS FOUR WORDS:
672
673 case 3:
674 dst_i++;
675 val >>= (BITS_PER_DIGIT-left_shift);
676 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
677 val >>= BITS_PER_DIGIT;
678 dst_p[dst_i++] = (sc_digit)(val & DIGIT_MASK);
679 val >>= BITS_PER_DIGIT;
680 dst_p[dst_i] = (sc_digit)val;
681 break;
682
683 }
684 return result;
685}

107// write the new value
108
109template< typename T, sc_writer_policy POL >
110inline
111void
112sc_buffer<T,POL>::write( const T& value_ )
113{
114 if( !base_type::policy_type::check_write(this,true) )
115 return;
116
117 this->m_new_val = value_;
118 this->request_update();
119}
120
121
122template< typename T, sc_writer_policy POL >
123inline
124void
125sc_buffer<T,POL>::update()
126{
127 base_type::policy_type::update();
128 base_type::do_update();
129}

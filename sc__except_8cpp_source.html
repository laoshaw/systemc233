38sc_unwind_exception::sc_unwind_exception( sc_process_b* proc_p, bool is_reset )
39 : m_proc_p(proc_p), m_is_reset( is_reset )
40{
41 sc_assert( m_proc_p );
42 m_proc_p->start_unwinding();
43}
44
45bool
46sc_unwind_exception::active() const
47{
48 return m_proc_p && m_proc_p->is_unwinding();
49}
50
51void
52sc_unwind_exception::clear() const
53{
54 sc_assert( m_proc_p );
55 m_proc_p->clear_unwinding();
56}
57
58const char*
59sc_unwind_exception::what() const SC_NOEXCEPT_
60{
61 return ( m_is_reset ) ? "RESET" : "KILL";
62}

64sc_unwind_exception::~sc_unwind_exception() SC_NOEXCEPT_
65{
66 if( active() ) {
67 // can't throw an exception, since we're already throwing
68 // -> abort instead
69 SC_REPORT_FATAL( SC_ID_RETHROW_UNWINDING_, m_proc_p->name() );
70 sc_abort(); // can't recover from here
71 }
72}

74// handle and translate uncaught exceptions here
75//
76// These exceptions can either escape from sc_main() directly,
77// indirectly from an SC_METHOD(), or are thrown from within
78// an SC_(C)THREAD()
79//
80// returns a pointer to a dynamically allocated sc_report object,
81// containing the caught message
82
83SC_API sc_report*
84sc_handle_exception()
85{
86 try {
87 // re-throw exception here
88 try { throw; }
89
90 catch( sc_report & ) // to be on the safe side
91 {
92 throw; // continue
93 }
94 catch( sc_unwind_exception const & )
95 {
96 SC_REPORT_ERROR( SC_ID_INTERNAL_ERROR_,
97 "unhandled kill/reset, should never happen" );
98 }
99 catch( std::exception const & x )
100 {
101 SC_REPORT_ERROR( SC_ID_SIMULATION_UNCAUGHT_EXCEPTION_, x.what() );
102 }
103 catch( char const * x )
104 {
105 SC_REPORT_ERROR( SC_ID_SIMULATION_UNCAUGHT_EXCEPTION_, x );
106 }
107 catch( ... )
108 {
109 SC_REPORT_ERROR( SC_ID_SIMULATION_UNCAUGHT_EXCEPTION_,
110 "UNKNOWN EXCEPTION" );
111 }
112 }
113 // everything is an sc_report now
114 catch( sc_report & rpt )
115 {
116 sc_report* rpt_p = new sc_report;
117 rpt_p->swap( rpt );
118 return rpt_p;
119 }
120 return 0;
121}

37static int
38sc_time_compare( const void* p1, const void* p2 )
39{
40 const sc_time* t1 = static_cast<const sc_time*>( p1 );
41 const sc_time* t2 = static_cast<const sc_time*>( p2 );
42
43 if( *t1 < *t2 ) {
44 return 1;
45 } else if( *t1 > *t2 ) {
46 return -1;
47 } else {
48 return 0;
49 }
50}

52sc_event_queue::sc_event_queue( sc_module_name name_ )
53 : sc_module( name_ ),
54 m_ppq( 128, sc_time_compare ),
55 m_e( sc_event::kernel_event ),
56 m_change_stamp(0),
57 m_pending_delta(0)
58{
59 SC_METHOD( fire_event );
60 sensitive << m_e;
61 dont_initialize();
62}
63
64sc_event_queue::~sc_event_queue()
65{
66 while (m_ppq.size() > 0) {
67 delete m_ppq.extract_top();
68 }
69}
70
71void sc_event_queue::cancel_all()
72{
73 m_pending_delta = 0;
74 while( m_ppq.size() > 0 )
75 delete m_ppq.extract_top();
76 m_e.cancel();
77}
78
79void sc_event_queue::notify (const sc_time& when)
80{
81 m_change_stamp = simcontext()->change_stamp();
82 sc_time* t = new sc_time( when+sc_time_stamp() );
83 if ( m_ppq.size()==0 || *t < *m_ppq.top() ) {
84 m_e.notify( when );
85 }
86 m_ppq.insert( t );
87}
88
89void sc_event_queue::fire_event()
90{
91 if ( m_ppq.empty() ) { // event has been cancelled
92 return;
93 }
94 sc_time* t = m_ppq.extract_top();
95 sc_assert( *t==sc_time_stamp() );
96 delete t;
97
98 if ( m_ppq.size() > 0 ) {
99 m_e.notify( *m_ppq.top() - sc_time_stamp() );
100 }
101}

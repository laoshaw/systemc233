145// ----------------------------------------------------------------------------
146// CLASS : sc_prim_channel_registry::async_update_list
147//
148// Thread-safe list of pending external updates
149// FOR INTERNAL USE ONLY!
150// ----------------------------------------------------------------------------
151
152class sc_prim_channel_registry::async_update_list
153{
154#ifndef SC_DISABLE_ASYNC_UPDATES
155public:
156
157 bool pending() const
158 {
159 return m_push_queue.size() != 0;
160 }
161
162 void suspend()
163 {
164 if( m_has_suspending_channels ) {
165 m_suspend_semaphore.wait();
166 m_suspend_semaphore.post(); // replace token
167 }
168 }
169
170 void append( sc_prim_channel& prim_channel_ )
171 {
172 sc_scoped_lock lock( m_mutex );
173 m_push_queue.push_back( &prim_channel_ );
174 m_suspend_semaphore.post();
175 // return releases the mutex
176 }
177
178 void accept_updates()
179 {
180 sc_assert( ! m_pop_queue.size() );
181 {
182 sc_scoped_lock lock( m_mutex );
183 m_push_queue.swap( m_pop_queue );
184 // leaving the block releases the mutex
185 }
186
187 std::vector< sc_prim_channel* >::const_iterator
188 it = m_pop_queue.begin(), end = m_pop_queue.end();
189 while( it!= end )
190 {
191 // we use request_update instead of perform_update
192 // to skip duplicates
193 (*it++)->request_update();
194 int sem_trywait = m_suspend_semaphore.trywait(); // this must never block !
195 sc_assert( sem_trywait == 0 );
196 }
197 m_pop_queue.clear();
198 }
199
200 bool attach_suspending( sc_prim_channel& p )
201 {
202 sc_scoped_lock lock( m_mutex );
203 std::vector<sc_prim_channel*>::iterator it =
204 std::find(m_suspending_channels.begin(), m_suspending_channels.end(), &p);
205 if ( it == m_suspending_channels.end() ) {
206 m_suspending_channels.push_back(&p);
207 m_has_suspending_channels = true;
208 return true;
209 }
210 return false;
211 // return releases the mutex
212 }
213
214 bool detach_suspending( sc_prim_channel& p )
215 {
216 sc_scoped_lock lock( m_mutex );
217 std::vector<sc_prim_channel*>::iterator it =
218 std::find(m_suspending_channels.begin(), m_suspending_channels.end(), &p);
219 if ( it != m_suspending_channels.end() ) {
220 *it = m_suspending_channels.back();
221 m_suspending_channels.pop_back();
222 m_has_suspending_channels = (m_suspending_channels.size() > 0);
223 return true;
224 }
225 return false;
226 // return releases the mutex
227 }
228
229 async_update_list() : m_has_suspending_channels() {}
230
231private:
232 sc_host_mutex m_mutex;
233 sc_host_semaphore m_suspend_semaphore;
234 std::vector< sc_prim_channel* > m_push_queue;
235 std::vector< sc_prim_channel* > m_pop_queue;
236 std::vector< sc_prim_channel* > m_suspending_channels;
237 bool m_has_suspending_channels;
238
239#endif // ! SC_DISABLE_ASYNC_UPDATES
240};

249void
250sc_prim_channel_registry::insert( sc_prim_channel& prim_channel_ )
251{
252 if( sc_is_running() ) {
253 SC_REPORT_ERROR( SC_ID_INSERT_PRIM_CHANNEL_, "simulation running" );
254 return;
255 }
256
257 if( m_simc->elaboration_done() ) {
258 SC_REPORT_ERROR( SC_ID_INSERT_PRIM_CHANNEL_, "elaboration done" );
259 return;
260 }
261
262#ifdef DEBUG_SYSTEMC
263 // check if prim_channel_ is already inserted
264 for( int i = 0; i < size(); ++ i ) {
265 if( &prim_channel_ == m_prim_channel_vec[i] ) {
266 SC_REPORT_ERROR( SC_ID_INSERT_PRIM_CHANNEL_, "already inserted" );
267 return;
268 }
269 }
270#endif
271
272 // insert
273 m_prim_channel_vec.push_back( &prim_channel_ );
274
275}
276
277void
278sc_prim_channel_registry::remove( sc_prim_channel& prim_channel_ )
279{
280 int i;
281 for( i = 0; i < size(); ++ i ) {
282 if( &prim_channel_ == m_prim_channel_vec[i] ) {
283 break;
284 }
285 }
286 if( i == size() ) {
287 SC_REPORT_ERROR( SC_ID_REMOVE_PRIM_CHANNEL_, 0 );
288 return;
289 }
290
291 // remove
292 m_prim_channel_vec[i] = m_prim_channel_vec.back();
293 m_prim_channel_vec.pop_back();
294
295#ifndef SC_DISABLE_ASYNC_UPDATES
296 // remove, if async suspending channel
297 m_async_update_list_p->detach_suspending(prim_channel_);
298#endif
299}

354// +----------------------------------------------------------------------------
355// |"sc_prim_channel_registry::perform_update"
356// |
357// | This method updates the values of the primitive channels in its update
358// | lists.
359// +----------------------------------------------------------------------------
360void
361sc_prim_channel_registry::perform_update()
362{
363 // Update the values for the primitive channels set external to the
364 // simulator.
365
366#ifndef SC_DISABLE_ASYNC_UPDATES
367 if( m_async_update_list_p->pending() )
368 m_async_update_list_p->accept_updates();
369#endif
370
371 sc_prim_channel* next_p; // Next update to perform.
372 sc_prim_channel* now_p; // Update now performing.
373
374 // Update the values for the primitive channels in the simulator's list.
375
376 now_p = m_update_list_p;
377 m_update_list_p = (sc_prim_channel*)sc_prim_channel::list_end;
378 for ( ; now_p != (sc_prim_channel*)sc_prim_channel::list_end;
379 now_p = next_p )
380 {
381 next_p = now_p->m_update_next_p;
382 now_p->perform_update();
383 }
384}

416

418 m_prim_channel_vec[m_construction_done]->construction_done();
419 }

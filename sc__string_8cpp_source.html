60class SC_API sc_string_rep
61{
62 friend class sc_string_old;
63 friend ::std::ostream& operator<<( ::std::ostream&, const sc_string_old& );
64 friend ::std::istream& operator>>( ::std::istream&, sc_string_old& );
65 friend sc_string_old operator+( const char*, const sc_string_old& );
66
67 sc_string_rep( int size = 16 ) :
68 ref_count(1), alloc( sc_roundup( size, 16 ) ), str( new char[alloc] )
69 {
70 *str = '\0';
71 }
72
73 sc_string_rep( const char* s ) : ref_count(1), alloc(0), str(0)
74 {
75 if (s) {
76 alloc = 1 + strlen(s);
77 str = strcpy( new char[alloc], s );
78 }
79 else {
80 alloc = 16;
81 str = strcpy( new char[alloc], "" );
82 }
83 }
84
85 sc_string_rep( const char* s, int n); // get first n chars from the string
86
87 ~sc_string_rep()
88 {
89 sc_assert( ref_count == 0 );
90 delete[] str;
91 }
92
93 void resize( int new_size );
94 void set_string( const char* s );
95
96 int ref_count;
97 int alloc;
98 char* str;
99};

104sc_string_rep::sc_string_rep( const char* s, int n) :
105 ref_count(1), alloc(0), str(0)
106{
107 if (s && n>0) {
108 alloc = 1 + n;
109 str = strncpy( new char[alloc], s,n );
110 str[n] = 00;
111 }
112 else {
113 alloc = 16;
114 str = strcpy( new char[alloc], "" );
115 }
116}

118void
119sc_string_rep::resize( int new_size )
120{
121 if (new_size <= alloc) return;
122 alloc = sc_roundup( new_size, 16 );
123 char* new_str = strcpy( new char[alloc], str );
124 delete[] str;
125 str = new_str;
126}

128void
129sc_string_rep::set_string( const char* s )
130{
131 int len = strlen(s);
132 resize( len + 1 );
133 strcpy( str, s );
134}

381sc_string_old sc_string_old::to_string(const char* format, ...)
382{
383 std::va_list argptr;
384 sc_string_old result;
385 char buffer[1024]; // static string buffer
386 buffer[1023]=000;
387
388 va_start(argptr, format);
389 int cnt = sc_vsnprintf(buffer, 1024, format, argptr);
390 if(cnt>1023) // string too long
391 {
392 int buf_size = 1024;
393 const int max_size = 65000;
394 char* buf = 0; // dynamic string buffer
395 do
396 {
397 delete[] buf;
398 buf_size*=2;
399 buf = new char[buf_size];
400 cnt = sc_vsnprintf(buf, buf_size, format, argptr);
401 }
402 while( buf_size<max_size && cnt>=buf_size);
403 if(cnt>=buf_size)
404 {
405 // string is longer the the maximum buffer size (max_size)
406 SC_REPORT_WARNING( sc_core::SC_ID_STRING_TOO_LONG_, "truncated" );
407 buf[buf_size-1] = 000;
408 }
409 result = buf;
410 delete[] buf;
411 }
412 else
413 result = buffer;
414
415 va_end(argptr);
416
417 return result;
418}

434// TODO: conveniece formatting functions for common types
435// e.g. sc_string_old("a=%d, s is %s").fmt(1).fmt("string")
436// should produce a=1, s is string
437// it should be safe: if less arguments specified
438// it should print %specifier; extra arguments should be ignored
439// if the type of the argument is incompatible with format
440// specifier it should be ignored
441//
442
443unsigned
444sc_string_old::fmt_length()const
445{
446 unsigned result=0;
447 if((*this)[0]!='%')
448 return 0;
449 else
450 result++;
451 if(is_delimiter("-+0 #",result)) // flags
452 result++;
453 while(is_delimiter("0123456789*",result)) // width
454 result++;
455 if(rep->str[result]=='.') // precision
456 {
457 result++;
458 unsigned old_result = result;
459 while(is_delimiter("0123456789*",result)) result++;
460 if(old_result == result) //error in format
461 return 0;
462 }
463 if(is_delimiter("hlL",result)) result++; // I64 is not supported
464 if(is_delimiter("cCdiouxXeEfgGnpsS",result))
465 result++;
466 else // error in format
467 return 0;
468 return result;
469}

477int
478sc_string_old::pos( const sc_string_old& sub_string ) const
479{
480 int sub_len = sub_string.length();
481 if( sub_len == 0 ) {
482 return 0; // empty string always matches
483 }
484 int ind = 0;
485 int len = length();
486 bool found = false;
487 while( ind < len && ! found )
488 {
489 found = ( sub_string == substr( ind, ind + sub_len - 1 ) );
490 ++ ind;
491 }
492 if( found ) {
493 return -- ind;
494 } else {
495 return -1;
496 }
497}

566
567::std::istream&
568operator >> ( ::std::istream& is, sc_string_old& s )
569{
570 if( s.rep->ref_count > 1 ) {
571 -- s.rep->ref_count;
572 s.rep = new sc_string_rep;
573 }
574
575 int i = 0;
576 char* p = s.rep->str;
577 char c;
578
579 // skip white spaces
580 while( is.get( c ) && isspace( c ) )
581 ;
582
583 for( ; is.good() && ! isspace( c ); is.get( c ) ) {
584 if( i > s.rep->alloc - 2 ) {
585 s.rep->str[i] = '\0';
586 s.rep->resize( (int) (s.rep->alloc * 1.5) );
587 p = s.rep->str + i;
588 }
589 *p ++ = c;
590 i ++;
591 }
592 *p = '\0';
593
594 return is;
595}

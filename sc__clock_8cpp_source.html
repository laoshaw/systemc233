205//------------------------------------------------------------------------------
206//"sc_clock::before_end_of_elaboration"
207//
208// This callback is used to spawn the edge processes for this object instance.
209// The processes are created here rather than the constructor for the object
210// so that the processes are registered with the global simcontext rather
211// than the scope of the clock's parent.
212//------------------------------------------------------------------------------
213#if ( defined(_MSC_VER) && _MSC_VER < 1300 ) //VC++6.0 doesn't support sc_spawn with functor.
214# define sc_clock_posedge_callback(ptr) sc_clock_posedge_callback
215
216# define sc_clock_negedge_callback(ptr) sc_clock_negedge_callback
217
218# define sc_spawn(a,b,c) { \
219 sc_process_handle result(new sc_spawn_object<a>(a(this),b,c)); \
220 }
221#endif // ( defined(_MSC_VER) && _MSC_VER < 1300 )
222
223void sc_clock::before_end_of_elaboration()
224{
225 std::string gen_base;
226 sc_spawn_options posedge_options; // Options for posedge process.
227 sc_spawn_options negedge_options; // Options for negedge process.
228
229 posedge_options.spawn_method();
230 posedge_options.dont_initialize();
231 posedge_options.set_sensitivity(&m_next_posedge_event);
232 gen_base = basename();
233 gen_base += "_posedge_action";
234 sc_spawn(sc_clock_posedge_callback(this),
235 sc_gen_unique_name( gen_base.c_str() ), &posedge_options);
236
237 negedge_options.spawn_method();
238 negedge_options.dont_initialize();
239 negedge_options.set_sensitivity(&m_next_negedge_event);
240 gen_base = basename();
241 gen_base += "_negedge_action";
242 sc_spawn( sc_clock_negedge_callback(this),
243 sc_gen_unique_name( gen_base.c_str() ), &negedge_options );
244}

296void
297sc_clock::init( const sc_time& period_,
298 double duty_cycle_,
299 const sc_time& start_time_,
300 bool posedge_first_ )
301{
302 if( period_ == SC_ZERO_TIME ) {
303 report_error( SC_ID_CLOCK_PERIOD_ZERO_, "increase the period" );
304 // may continue, if suppressed
305 }
306 m_period = period_;
307 m_posedge_first = posedge_first_;
308
309 if( duty_cycle_ <= 0.0 || duty_cycle_ >= 1.0 ) {
310 m_duty_cycle = 0.5;
311 } else {
312 m_duty_cycle = duty_cycle_;
313 }
314
315 m_negedge_time = m_period * m_duty_cycle;
316 m_posedge_time = m_period - m_negedge_time;
317
318 if( m_negedge_time == SC_ZERO_TIME ) {
319 report_error( SC_ID_CLOCK_HIGH_TIME_ZERO_,
320 "increase the period or increase the duty cycle" );
321 // may continue, if suppressed
322 }
323 if( m_posedge_time == SC_ZERO_TIME ) {
324 report_error( SC_ID_CLOCK_LOW_TIME_ZERO_,
325 "increase the period or decrease the duty cycle" );
326 // may continue, if suppressed
327 }
328
329 if( posedge_first_ ) {
330 this->m_cur_val = false;
331 this->m_new_val = false;
332 } else {
333 this->m_cur_val = true;
334 this->m_new_val = true;
335 }
336
337 m_start_time = start_time_;
338
339}

42// to avoid code bloat in sc_signal<T>
43
44void
45sc_signal_invalid_writer( sc_object* target, sc_object* first_writer,
46 sc_object* second_writer, bool check_delta )
47{
48 if ( second_writer )
49 {
50 std::stringstream msg;
51 msg
52 << "\n signal "
53 "`" << target->name() << "' "
54 "(" << target->kind() << ")"
55 << "\n first driver "
56 "`" << first_writer->name() << "' "
57 "(" << first_writer->kind() << ")"
58 << "\n second driver "
59 "`" << second_writer->name() << "' "
60 "(" << second_writer->kind() << ")";
61
62 if( check_delta )
63 {
64 msg << "\n conflicting write in delta cycle "
65 << sc_delta_count();
66 }
67 SC_REPORT_ERROR( SC_ID_MORE_THAN_ONE_SIGNAL_DRIVER_,
68 msg.str().c_str() );
69 }
70}

78bool
79sc_writer_policy_check_port::
80 check_port( sc_object* target, sc_port_base * port_, bool is_output )
81{
82 if ( is_output && sc_get_curr_simcontext()->write_check() &&
83 !sc_get_curr_simcontext()->write_check_conflicts_only() )
84 {
85 // an out or inout port; only one can be connected
86 if( m_output != 0) {
87 sc_signal_invalid_writer( target, m_output, port_, false );
88 return false;
89 } else {
90 m_output = port_;
91 }
92 }
93 return true;
94}

166template< sc_writer_policy POL >
167void
168sc_signal<bool,POL>::update()
169{
170 policy_type::update();
171 if( !( base_type::m_new_val == base_type::m_cur_val ) ) {
172 do_update();
173 }
174}
175
176template< sc_writer_policy POL >
177void
178sc_signal<bool,POL>::do_update()
179{
180 // order of execution below is important, the notify_processes() call
181 // must come after the update of m_cur_val for things to work properly!
182 base_type::do_update();
183 if ( m_reset_p ) m_reset_p->notify_processes();
184
185 sc_event* event_p = this->m_cur_val ? m_posedge_event_p : m_negedge_event_p;
186 sc_signal_channel::notify_next_delta( event_p );
187}

242sc_signal<sc_logic,POL>::do_update()
243{
244 base_type::do_update();
245
246 if( this->m_cur_val == sc_dt::SC_LOGIC_1 ) {
247 sc_signal_channel::notify_next_delta( m_posedge_event_p );
248 }
249 else if( this->m_cur_val == sc_dt::SC_LOGIC_0 ) {
250 sc_signal_channel::notify_next_delta( m_negedge_event_p );
251 }

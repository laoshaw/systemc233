259inline
260sc_bv_base::value_type
261sc_bv_base::get_bit( int i ) const
262{
263 int wi = i / SC_DIGIT_SIZE;
264 int bi = i % SC_DIGIT_SIZE;
265 return value_type( (m_data[wi] >> bi) & SC_DIGIT_ONE );
266}
267
268inline
269void
270sc_bv_base::set_bit( int i, value_type value )
271{
272 int wi = i / SC_DIGIT_SIZE;
273 int bi = i % SC_DIGIT_SIZE;
274 sc_digit mask = SC_DIGIT_ONE << bi;
275 m_data[wi] |= mask; // set bit to 1
276 m_data[wi] &= value << bi | ~mask;
277}

290inline
291void
292sc_bv_base::clean_tail()
293{
294 int wi = m_size - 1;
295 int bi = m_len % SC_DIGIT_SIZE;
296 if ( bi != 0 ) m_data[wi] &= ~SC_DIGIT_ZERO >> (SC_DIGIT_SIZE - bi);
297}

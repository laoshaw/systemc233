47std::vector< sc_object* > const &
48sc_vector_base::get_elements() const
49{
50 if( !objs_vec_ )
51 objs_vec_ = new std::vector< sc_object* >;
52
53 if( objs_vec_->size() || !size() )
54 return *objs_vec_;
55
56 objs_vec_->reserve( size() );
57 for( const_iterator it=begin(); it != end(); ++it )
58 if( sc_object* obj = object_cast(*it) )
59 objs_vec_->push_back( obj );
60
61 return *objs_vec_;
62}

71void
72sc_vector_base::check_index( size_type i ) const
73{
74 if( i>=size() )
75 {
76 std::stringstream str;
77 str << name()
78 << "[" << i << "] >= size() = " << size();
79 SC_REPORT_ERROR( SC_ID_OUT_OF_BOUNDS_, str.str().c_str() );
80 sc_abort(); // can't recover from here
81 }
82}

84bool
85sc_vector_base::check_init( size_type n ) const
86{
87 if ( !n )
88 return false;
89
90 if( size() ) // already filled
91 {
92 std::stringstream str;
93 str << name()
94 << ", size=" << size()
95 << ", requested size=" << n;
96 SC_REPORT_ERROR( SC_ID_VECTOR_INIT_CALLED_TWICE_
97 , str.str().c_str() );
98 return false;
99 }
100 return true;
101}

105{

110
111 if( !size() ) {
112 str << "not initialised yet";
113 } else if ( dst_empty_ ) {
114 str << "empty range given";
115 } else {
116 str << "empty destination range given";

118
119 SC_REPORT_WARNING( SC_ID_VECTOR_BIND_EMPTY_, str.str().c_str() );
120}

133sc_vector_base::context_scope::context_scope( sc_vector_base* owner )
134 : owner_(NULL)
135{
136 sc_simcontext* simc = owner->simcontext();
137 sc_assert( simc == sc_get_curr_simcontext() );
138
139 sc_object* parent = owner->get_parent_object();
140 sc_object* active = simc->active_object();
141
142 if (parent != active) // override object creation context
143 {
144 owner_ = owner;
145 owner->simcontext()->get_object_manager()
146 ->hierarchy_push( owner_->get_parent_object() );
147 }
148}
149
150sc_vector_base::context_scope::~context_scope()
151{
152 if (owner_) // restore current object context
153 {
154 sc_object* obj = owner_->simcontext()->get_object_manager()->hierarchy_pop();
155 sc_assert( obj == owner_->get_parent_object() );
156 }
157}

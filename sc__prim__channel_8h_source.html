229// ----------------------------------------------------------------------------
230// CLASS : sc_prim_channel_registry
231//
232// Registry for all primitive channels.
233// FOR INTERNAL USE ONLY!
234// ----------------------------------------------------------------------------
235
236class sc_prim_channel_registry
237{
238 friend class sc_simcontext;
239
240public:
241
242 void insert( sc_prim_channel& );
243 void remove( sc_prim_channel& );
244
245
246 int size() const
247 { return static_cast<int>(m_prim_channel_vec.size()); }
248
249 inline void request_update( sc_prim_channel& );
250 void async_request_update( sc_prim_channel& );
251
252 bool pending_updates() const
253 {
254 return m_update_list_p != (sc_prim_channel*)sc_prim_channel::list_end
255 || pending_async_updates();
256 }
257
258 bool pending_async_updates() const;
259
260 // synchronization with attached async suspending channels
261 // - potentially blocks the current thread, if no explicitly
262 // attached async channels have posted updates, yet
263 // - returns true, if and only if there are NO pending synchronous
264 // updates after resuming from the external synchronization
265 bool async_suspend();
266
267 // (un)register a channel as being asynchronous
268 // - presence of asynchronous channels leads async_suspend() to
269 // block until any external async updates have been received
270 // (instead of exiting the simulation upon starvation)
271 bool async_attach_suspending(sc_prim_channel&);
272 bool async_detach_suspending(sc_prim_channel&);
273
274private:
275
276 // constructor
277 explicit sc_prim_channel_registry( sc_simcontext& simc_ );
278
279 // destructor
280 ~sc_prim_channel_registry();
281
282 // called during the update phase of a delta cycle
283 void perform_update();
284
285 // called when construction is done
286 bool construction_done();
287
288 // called when elaboration is done
289 void elaboration_done();
290
291 // called before simulation starts
292 void start_simulation();
293
294 // called after simulation ends
295 void simulation_done();
296
297 // disabled
298 sc_prim_channel_registry();
299 sc_prim_channel_registry( const sc_prim_channel_registry& );
300 sc_prim_channel_registry& operator = ( const sc_prim_channel_registry& );
301
302private:
303 class async_update_list;
304
305 async_update_list* m_async_update_list_p; // external updates.
306 int m_construction_done; // # of constructs.
307 std::vector<sc_prim_channel*> m_prim_channel_vec; // existing channels.
308 sc_simcontext* m_simc; // simulator context.
309 sc_prim_channel* m_update_list_p; // internal updates.
310};

341{
342 if( ! m_update_next_p ) {

60// resolves sc_dt::sc_lv<W> values and returns the resolved value
61
62template <int W>
63inline
64void
65sc_lv_resolve<W>::resolve( sc_dt::sc_lv<W>& result_,
66 const std::vector<sc_dt::sc_lv<W>*>& values_ )
67{
68 int sz = values_.size();
69
70 sc_assert( sz != 0 );
71
72 if( sz == 1 ) {
73 result_ = *values_[0];
74 return;
75 }
76
77 for( int j = result_.length() - 1; j >= 0; -- j ) {
78 sc_dt::sc_logic_value_t res = (*values_[0])[j].value();
79 for( int i = sz - 1; i > 0 && res != 3; -- i ) {
80 res = sc_logic_resolution_tbl[res][(*values_[i])[j].value()];
81 }
82 result_[j] = res;
83 }
84}

182// write the new value
183
184template <int W>
185inline
186void
187sc_signal_rv<W>::write( const value_type& value_ )
188{
189 sc_process_b* cur_proc = sc_get_current_process_b();
190
191 bool value_changed = false;
192 bool found = false;
193
194 for( int i = m_proc_vec.size() - 1; i >= 0; -- i ) {
195 if( cur_proc == m_proc_vec[i] ) {
196 if( value_ != *m_val_vec[i] ) {
197 *m_val_vec[i] = value_;
198 value_changed = true;
199 }
200 found = true;
201 break;
202 }
203 }
204
205 if( ! found ) {
206 m_proc_vec.push_back( cur_proc );
207 m_val_vec.push_back( new value_type( value_ ) );
208 value_changed = true;
209 }
210
211 if( value_changed ) {
212 this->request_update();
213 }
214}

223 base_type::update();

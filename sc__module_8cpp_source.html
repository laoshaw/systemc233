57// ----------------------------------------------------------------------------
58// CLASS : sc_module_dynalloc_list
59//
60// Garbage collection for modules dynamically allocated with SC_NEW.
61// ----------------------------------------------------------------------------
62
63class sc_module_dynalloc_list
64{
65public:
66
67 sc_module_dynalloc_list() : m_list()
68 {}
69
70 ~sc_module_dynalloc_list();
71
72 void add( sc_module* p )
73 { m_list.push_back( p ); }
74
75private:
76
77 sc_plist<sc_module*> m_list;
78};
79
80
81//------------------------------------------------------------------------------
82//"~sc_module_dynalloc_list"
83//
84// Note we clear the m_parent field for the module being deleted. This because
85// we process the list front to back so the parent has already been deleted,
86// and we don't want ~sc_object() to try to access the parent which may
87// contain garbage.
88//------------------------------------------------------------------------------
89sc_module_dynalloc_list::~sc_module_dynalloc_list()
90{
91 sc_plist<sc_module*>::iterator it( m_list );
92 while( ! it.empty() ) {
93 (*it)->m_parent = 0;
94 delete *it;
95 it ++;
96 }
97}

102SC_API sc_module*
103sc_module_dynalloc( sc_module* module_ )
104{
105 static sc_module_dynalloc_list dynalloc_list;
106 dynalloc_list.add( module_ );
107 return module_;
108}

143void
144sc_module::sc_module_init()
145{
146 simcontext()->get_module_registry()->insert( *this );
147 simcontext()->hierarchy_push( this );
148 m_end_module_called = false;
149 m_module_name_p = 0;
150 m_port_vec = new std::vector<sc_port_base*>;
151 m_port_index = 0;
152}
153
154/*
155 * This form of the constructor assumes that the user has
156 * used an sc_module_name parameter for his/her constructor.
157 * That parameter object has been pushed onto the stack,
158 * and can be looked up by calling the
159 * top_of_module_name_stack() function of the object manager.
160 * This technique has two advantages:
161 *
162 * 1) The user no longer has to write sc_module(name) in the
163 * constructor initializer.
164 * 2) The user no longer has to call end_module() at the end
165 * of the constructor -- a common negligence.
166 *
167 * But it is important to note that sc_module_name may be used
168 * in the user's constructor's parameter. If it is used anywhere
169 * else, unexpected things will happen. The error-checking
170 * mechanism builtin here cannot hope to catch all misuses.
171 *
172 */
173
174sc_module::sc_module()
175: sc_object(::sc_core::sc_get_curr_simcontext()
176 ->get_object_manager()
177 ->top_of_module_name_stack()
178 ->operator const char*()),
179 sensitive(this),
180 sensitive_pos(this),
181 sensitive_neg(this),
182 m_end_module_called(false),
183 m_port_vec(),
184 m_port_index(0),
185 m_name_gen(0),
186 m_module_name_p(0)
187{
188 /* When this form is used, we better have a fresh sc_module_name
189 on the top of the stack */
190 sc_module_name* mod_name =
191 simcontext()->get_object_manager()->top_of_module_name_stack();
192 if (0 == mod_name || 0 != mod_name->m_module_p) {
193 SC_REPORT_ERROR( SC_ID_SC_MODULE_NAME_REQUIRED_, 0 );
194 sc_abort(); // can't recover from here
195 }
196 sc_module_init();
197 mod_name->set_module( this );
198 m_module_name_p = mod_name; // must come after sc_module_init call.
199}
200
201sc_module::sc_module( const sc_module_name& )
202: sc_object(::sc_core::sc_get_curr_simcontext()
203 ->get_object_manager()
204 ->top_of_module_name_stack()
205 ->operator const char*()),
206 sensitive(this),
207 sensitive_pos(this),
208 sensitive_neg(this),
209 m_end_module_called(false),
210 m_port_vec(),
211 m_port_index(0),
212 m_name_gen(0),
213 m_module_name_p(0)
214{
215 /* For those used to the old style of passing a name to sc_module,
216 this constructor will reduce the chance of making a mistake */
217
218 /* When this form is used, we better have a fresh sc_module_name
219 on the top of the stack */
220 sc_module_name* mod_name =
221 simcontext()->get_object_manager()->top_of_module_name_stack();
222 if (0 == mod_name || 0 != mod_name->m_module_p) {
223 SC_REPORT_ERROR( SC_ID_SC_MODULE_NAME_REQUIRED_, 0 );
224 sc_abort(); // can't recover from here
225 }
226 sc_module_init();
227 mod_name->set_module( this );
228 m_module_name_p = mod_name; // must come after sc_module_init call.
229}

269sc_module::~sc_module()
270{
271 delete m_port_vec;
272 delete m_name_gen;
273 orphan_child_objects();
274 if ( m_module_name_p )
275 {
276 m_module_name_p->clear_module( this ); // must be before end_module()
277 end_module();
278 }
279 simcontext()->get_module_registry()->remove( *this );
280}

315void
316sc_module::end_module()
317{
318 if( ! m_end_module_called ) {
319 /* TBD: Can check here to alert the user that end_module
320 was not called for a previous module. */
321 (void)sc_get_curr_simcontext()->hierarchy_pop();
322 sc_get_curr_simcontext()->reset_curr_proc();
323 sensitive.reset();
324 sensitive_pos.reset();
325 sensitive_neg.reset();
326 m_end_module_called = true;
327 m_module_name_p = 0; // make sure we are not called in ~sc_module().
328 }
329}

445void
446sc_module::set_stack_size( std::size_t size )
447{
448 sc_process_handle proc_h(
449 sc_is_running() ?
450 sc_get_current_process_handle() :
451 sc_get_last_created_process_handle()
452 );
453 sc_thread_handle thread_h; // Current process as thread.
454
455
456 thread_h = (sc_thread_handle)proc_h;
457 if ( thread_h )
458 {
459 thread_h->set_stack_size( size );
460 }
461 else
462 {
463 SC_REPORT_WARNING( SC_ID_SET_STACK_SIZE_, 0 );
464 }
465}

507void
508sc_module::positional_bind( sc_interface& interface_ )
509{
510 if( m_port_index == (int)m_port_vec->size() ) {
511 std::stringstream msg;
512 if( m_port_index == 0 ) {
513 msg << "module `" << name() << "' has no ports";
514 } else {
515 msg << "all ports of module `" << name() << "' are bound";
516 }
517 SC_REPORT_ERROR( SC_ID_BIND_IF_TO_PORT_, msg.str().c_str() );
518 return;
519 }
520 int status = (*m_port_vec)[m_port_index]->pbind( interface_ );
521 if( status != 0 ) {
522 std::stringstream msg;
523 switch( status ) {
524 case 1:
525 msg << "port " << m_port_index
526 << " of module `" << name() << "' is already bound";
527 break;
528 case 2:
529 msg << "type mismatch on port " << m_port_index
530 << " of module `" << name() << "'";
531 break;
532 default:
533 msg << "unknown error";
534 break;
535 }
536 SC_REPORT_ERROR( SC_ID_BIND_IF_TO_PORT_, msg.str().c_str() );
537 }
538 ++ m_port_index;
539}

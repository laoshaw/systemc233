148void
149sc_export_registry::insert( sc_export_base* export_ )
150{
151 if( sc_is_running() ) {
152 export_->report_error(SC_ID_INSERT_EXPORT_, "simulation running");
153 return;
154 }
155
156 if( m_simc->elaboration_done() ) {
157 export_->report_error(SC_ID_INSERT_EXPORT_, "elaboration done");
158 return;
159 }
160
161#ifdef DEBUG_SYSTEMC
162 // check if port_ is already inserted
163 for( int i = size() - 1; i >= 0; -- i ) {
164 if( export_ == m_export_vec[i] ) {
165 export_->report_error( SC_ID_INSERT_EXPORT_,
166 "export already inserted ");
167 return;
168 }
169 }
170#endif
171
172 sc_module* curr_module = m_simc->hierarchy_curr();
173 if( curr_module == 0 ) {
174 export_->report_error( SC_ID_EXPORT_OUTSIDE_MODULE_ );
175 return;
176 }
177
178 // insert
179 m_export_vec.push_back( export_ );
180}
181
182void
183sc_export_registry::remove( sc_export_base* export_ )
184{
185 if (size()==0) return;
186 int i;
187 for( i = size() - 1; i >= 0; -- i ) {
188 if( export_ == m_export_vec[i] ) {
189 break;
190 }
191 }
192 if( i == -1 ) {
193 export_->report_error( SC_ID_SC_EXPORT_NOT_REGISTERED_ );
194 return;
195 }
196
197 // remove
198 m_export_vec[i] = m_export_vec.back();
199 m_export_vec.pop_back();
200}

218// called when construction is done
219
220bool
221sc_export_registry::construction_done()
222{
223 if( m_construction_done == size() )
224 // nothing has been updated
225 return true;
226
227 for( int i = size()-1; i >= m_construction_done; --i ) {
228 m_export_vec[i]->construction_done();
229 }
230
231 m_construction_done = size();
232 return false;
233}

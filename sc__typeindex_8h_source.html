49class sc_type_index
50{
51public:
52 sc_type_index(const std::type_info& ti)
53 : info_(&ti)
54 {}
55
56 bool operator==( const sc_type_index& rhs ) const
57 { return *info_ == *rhs.info_; }
58
59 bool operator!=( const sc_type_index& rhs ) const
60 { return *info_ != *rhs.info_; }
61
62 bool operator< ( const sc_type_index& rhs ) const
63 { return info_->before( *rhs.info_ ); }
64
65 bool operator<=( const sc_type_index& rhs ) const
66 { return !(*this > rhs); }
67
68 bool operator> ( const sc_type_index& rhs ) const
69 { return rhs.info_->before( *info_ ); }
70
71 bool operator>=( const sc_type_index& rhs ) const
72 { return !(*this < rhs); }
73
74 const char* name() const
75 { return info_->name(); }
76
77private:
78 std::type_info const* info_;
79
80}; // class sc_type_index
